CREATE TABLE IF NOT EXISTS claude_insights (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  workspace_id UUID NOT NULL REFERENCES workspaces(id)
    ON DELETE CASCADE,

  -- Content
  insight_text TEXT NOT NULL,
  insight_type TEXT NOT NULL,
  -- 'pipeline' | 'forecast' | 'rep' | 'deal' | 'account'
  -- | 'strategic' | 'process' | 'competitive' | 'coaching'
  severity TEXT NOT NULL DEFAULT 'info',
  -- 'critical' | 'warning' | 'info' | 'positive'

  -- Source
  trigger_surface TEXT NOT NULL DEFAULT 'mcp',
  trigger_query TEXT,          -- the user question that produced this
  mcp_call_id UUID,            -- FK to mcp_calls if available
  tool_name TEXT,              -- which tool generated this

  -- Entity context (optional)
  entity_type TEXT,            -- 'deal' | 'rep' | 'account' | null
  entity_id UUID,
  entity_name TEXT,

  -- Lifecycle
  status TEXT NOT NULL DEFAULT 'active',
  -- 'active' | 'dismissed' | 'actioned'
  dismissed_at TIMESTAMPTZ,
  actioned_at TIMESTAMPTZ,

  -- Deduplication
  content_hash TEXT,           -- hash of insight_text for dedup
  CONSTRAINT uq_claude_insight_hash
    UNIQUE (workspace_id, content_hash),

  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_claude_insights_workspace
  ON claude_insights(workspace_id, created_at DESC);
CREATE INDEX idx_claude_insights_type
  ON claude_insights(workspace_id, insight_type, severity);
CREATE INDEX idx_claude_insights_entity
  ON claude_insights(workspace_id, entity_type, entity_id)

## Shared MCP Tool Interface

File: server/mcp/tools/types.ts  (new)

```typescript
export interface McpToolInput {
  save?: boolean    // default true — set false to skip persistence
  [key: string]: any
}

export interface McpToolResult {
  data: any
  saved: boolean    // whether this turn's output was persisted
  save_location?: string  // e.g. 'skill_runs', 'deliberation_runs'
  insight_id?: string    // if a claude_insight was created
}

// Helper used by all tool handlers
export async function maybeAutoSave(
  workspaceId: string,
  toolName: string,
  insightText: string,
  insightType: string,
  severity: string,
  triggerQuery: string,
  entityType?: string,
  entityId?: string,
  entityName?: string,
  save: boolean = true
): Promise<string | null> {
  if (!save) return null

  const contentHash = require('crypto')
    .createHash('md5')
    .update(insightText)
    .digest('hex')

  try {
    const result = await db.query(`
      INSERT INTO claude_insights
        (workspace_id, insight_text, insight_type, severity,
         trigger_surface, tool_name, entity_type, entity_id,
         entity_name, content_hash, trigger_query)
      VALUES ($1,$2,$3,$4,'mcp',$5,$6,$7,$8,$9,$10)
      ON CONFLICT (workspace_id, content_hash) DO NOTHING
      RETURNING id
    `, [
      workspaceId, insightText, insightType, severity,
      toolName, entityType ?? null, entityId ?? null,
      entityName ?? null, contentHash, triggerQuery,
    ])
    return result.rows[0]?.id ?? null
  } catch {
    return null  // never fail a tool call due to save error
  }
}

weekly-recap, project-recap`,
  inputSchema: {

        type: 'boolean',

      },

  handler: async (args, workspaceId) => {
    const save = args.save !== false

    // Dynamically register the skill if not registered
    await ensureSkillRegistered(args.skill_id)

    const skill = skillRegistry.get(args.skill_id)
    if (!skill) throw new Error(
      `Skill not found: ${args.skill_id}`
    )

    // Check for recent run (< 4 hours) to avoid redundant runs
    const recent = await db.query(`
      SELECT id, output, created_at
      FROM skill_runs
      WHERE workspace_id = $1
        AND skill_id = $2
        AND status = 'completed'
        AND created_at > NOW() - INTERVAL '4 hours'
      ORDER BY created_at DESC LIMIT 1
    `, [workspaceId, args.skill_id])

    let runId: string
    let output: any

    if (recent.rows.length) {
      runId = recent.rows[0].id
      output = recent.rows[0].output
    } else {
      const runtime = new SkillRuntime()
      const result = await runtime.executeSkill(
        skill, workspaceId, {}
      )
      runId = result.runId
      const run = await db.query(
        `SELECT output FROM skill_runs WHERE id = $1`,
        [runId]
      )
      output = run.rows[0]?.output
    }

    const claims = output?.evidence?.claims ?? []
    const narrative = output?.narrative ?? output?.summary ?? ''

    // Auto-save insight summary if save = true
    let insightId: string | null = null
    if (save && narrative) {
      insightId = await maybeAutoSave(
        workspaceId,
        args.skill_id,
        narrative.slice(0, 1000),
        mapSkillToInsightType(args.skill_id),
        getTopSeverity(claims),
        `run_skill: ${args.skill_id}`,
      )
    }

    return {
      skill_id: args.skill_id,
      run_id: runId,
      finding_count: claims.length,
      top_findings: claims.slice(0, 10).map((c: any) => ({
        finding: c.claim_text,
        severity: c.severity,
        metric: c.metric_name,
        value: c.metric_values?.[0],
      })),
      narrative: narrative.slice(0, 2000),
      saved: save,
      save_location: 'skill_runs',
      insight_id: insightId,
    }
  },
}

// Map skill ID to insight type
function mapSkillToInsightType(skillId: string): string {
  if (skillId.includes('pipeline')) return 'pipeline'
  if (skillId.includes('forecast')) return 'forecast'
  if (skillId.includes('rep') || skillId.includes('coaching'))
    return 'rep'
  if (skillId.includes('deal')) return 'deal'
  if (skillId.includes('icp') || skillId.includes('strategy'))
    return 'strategic'
  if (skillId.includes('competitive')) return 'competitive'
  if (skillId.includes('conversation') || skillId.includes('voice'))
    return 'coaching'
  return 'process'
}

function getTopSeverity(claims: any[]): string {
  if (claims.some(c => c.severity === 'critical')) return 'critical'
  if (claims.some(c => c.severity === 'warning')) return 'warning'
  return 'info'
}

### ensureSkillRegistered helper

The registry uses lazy registration. Build a helper that
imports and registers a skill by ID on demand:

```typescript
async function ensureSkillRegistered(skillId: string) {
  if (skillRegistry.get(skillId)) return

  try {
    // Dynamic import from skills/library/
    const skillFile = skillId.replace(/-/g, '-')
    const module = await import(
      `../../skills/library/${skillFile}`
    )
    const skill = module.default ?? Object.values(module)[0]
    if (skill) skillRegistry.register(skill)
  } catch {
    // Skill file not found — will throw "not found" in handler
  }
}

export const saveToReport: McpTool = {
  name: 'save_to_report',
  description: `Saves content to the current WBR or QBR report.
CALL THIS when the user asks to add something to a report, or 
when you generate content that belongs in the weekly brief.
Content appears immediately in the Report Viewer.`,
  inputSchema: {
    type: 'object',
    required: ['content', 'section_title'],
    properties: {
      content: {
        type: 'string',
        description: 'Markdown content to add to the section',
      },
      section_title: {
        type: 'string',
        description: 'Which section to add to. e.g. "Pipeline Health Snapshot", "Key Actions & Owners"',
      },
      document_id: {
        type: 'string',
        description: 'Specific document UUID. If omitted, uses most recent WBR.',
      },
      append: {
        type: 'boolean',
        description: 'Append to existing section content (default: true)',
        default: true,
      },
    },
  },
  handler: async (args, workspaceId) => {
    // Find document
    let docId = args.document_id
    if (!docId) {
      const latest = await db.query(`
        SELECT id FROM report_documents
        WHERE workspace_id = $1
          AND document_type IN ('wbr', 'qbr')
        ORDER BY generated_at DESC LIMIT 1
      `, [workspaceId])
      if (!latest.rows.length) throw new Error(
        'No WBR/QBR found. Generate one first with generate_report.'
      )
      docId = latest.rows[0].id
    }

    // Find or create the section
    const doc = await db.query(
      `SELECT sections FROM report_documents WHERE id = $1`,
      [docId]
    )
    const sections = doc.rows[0]?.sections ?? []
    const sectionIdx = sections.findIndex(
      (s: any) => (s.title ?? s.label ?? '')
        .toLowerCase()
        .includes(args.section_title.toLowerCase())
    )

    if (sectionIdx >= 0) {
      const existing = sections[sectionIdx].narrative ?? ''
      sections[sectionIdx].narrative = args.append !== false
        ? `${existing}\n\n${args.content}`.trim()
        : args.content
    } else {
      // Create new section
      sections.push({
        section_id: `mcp-${Date.now()}`,
        title: args.section_title,
        narrative: args.content,
        source_skills: [],
        confidence: 0.9,
      })
    }

    await db.query(
      `UPDATE report_documents SET sections = $1 WHERE id = $2`,
      [JSON.stringify(sections), docId]
    )

    return {
      saved: true,
      document_id: docId,
      section: args.section_title,
      view_url: `https://pandoragtm.com/reports/${docId}`,
    }
  },
}

    // Log to mcp_calls
    const callId = await logMcpCall(
      workspaceId, name, args, result, duration
    )

    // Auto-checkpoint: if tool returned structured data
    // and save was not explicitly false, emit a checkpoint
    // event that Claude can use to trigger save_claude_insight
    const shouldCheckpoint = args?.save !== false
      && isCheckpointWorthy(name, result)

    return {
      content: [{
        type: 'text',
        text: JSON.stringify({
          ...result,
          _pandora_meta: {
            call_id: callId,
            auto_saved: result.saved ?? false,
            checkpoint_suggested: shouldCheckpoint,
            // This hint tells Claude to call save_claude_insight
            // if it synthesizes something meaningful from this data
            save_hint: shouldCheckpoint
              ? 'If you synthesize a meaningful insight from this data, call save_claude_insight automatically.'
              : undefined,
          }
        }, null, 2),
      }]
    }
  } catch (err) {
    // ... error handling
  }
})

function isCheckpointWorthy(toolName: string, result: any): boolean {
  // Tools that produce meaningful insights worth auto-saving
  const CHECKPOINT_TOOLS = [
    'run_pipeline_hygiene', 'run_forecast_rollup',
    'run_deal_risk_review', 'run_rep_scorecard',
    'run_deliberation', 'run_skill',
    'run_conversation_intelligence', 'run_icp_discovery',
    'run_competitive_intelligence', 'run_strategy_insights',
    'run_bowtie_analysis', 'run_monte_carlo',
    'get_pipeline_health', 'get_forecast_rollup',
    'get_concierge_brief',
  ]
  return CHECKPOINT_TOOLS.includes(toolName)
    && result?.finding_count > 0
}
```

The `save_hint` in `_pandora_meta` is the mechanism. Claude
reads tool output and follows instructions embedded in it.
This is standard MCP behavior — the tool output can contain
instructions to Claude about what to do next.

## Opt-Out Mechanism

Every tool that auto-saves supports { save: false }:

```typescript
// User can say "don't save this" to Claude
// Claude passes save: false to the tool
run_pipeline_hygiene({ save: false })
run_deliberation({ mode: 'bull_bear', deal_id: '...', save: false })
save_claude_insight — this IS the save tool, no opt-out
```

For the automatic save_claude_insight call triggered by
save_hint in _pandora_meta: Claude will not call it if
the user says anything like "don't save", "just exploring",
"don't add this to Pandora". The tool description is explicit
about this:

```
Do NOT call it when: the user explicitly says not to save,
the user says they're just exploring or testing,
or the user says "don't add this to Pandora".
```
