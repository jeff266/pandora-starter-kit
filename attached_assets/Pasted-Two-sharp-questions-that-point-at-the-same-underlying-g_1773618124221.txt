function generateHypothesisRecommendation(hypothesis: StandingHypothesis): string {
  const isBreached = hypothesis.alert_direction === 'below'
    ? hypothesis.current_value < hypothesis.alert_threshold
    : hypothesis.current_value > hypothesis.alert_threshold

  const gap = Math.abs(hypothesis.current_value - hypothesis.alert_threshold)
  const gapPct = (gap / hypothesis.alert_threshold * 100).toFixed(0)

  if (isBreached) {
    // Should have sprint actions — this is a fallback
    return `Threshold breached. Run sprint assembly to generate specific actions.`
  }

  // Healthy — generate a monitoring recommendation
  const recommendations: Record<string, (h: StandingHypothesis) => string> = {
    pipeline_coverage_ratio: (h) =>
      `Coverage is ${gapPct}% above threshold. Confirm Q2 pipeline generation is in motion before quarter turn — coverage typically drops 0.3–0.5x in the first two weeks of a new quarter.`,

    large_deal_win_rate: (h) =>
      `Win rate is above threshold. Monitor: if any large deal (>$50K) goes stale this week, this hypothesis may trip. Check the conversion rate sprint actions — they overlap with this cohort.`,

    smb_sales_cycle_days: (h) =>
      `Cycle length is within threshold — no action required this week. Alert fires if average exceeds ${h.alert_threshold} days.`,

    // Default for unknown metrics
    default: (h) =>
      `${formatMetricValue(h.current_value, h.metric)} is within threshold (target: ${h.alert_direction === 'below' ? '≥' : '≤'} ${formatMetricValue(h.alert_threshold, h.metric)}). No action required this week.`,
  }

  const fn = recommendations[hypothesis.metric] ?? recommendations.default
  return fn(hypothesis)
}
