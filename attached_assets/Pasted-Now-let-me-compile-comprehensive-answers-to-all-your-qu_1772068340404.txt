  SELECT c.id, c.call_date, c.deal_id,         
    (SELECT MIN(a.activity_date) FROM          
  activities a

       AND a.activity_date > c.call_date

  INTERVAL '3 days') as next_activity

  HAVING next_activity IS NULL
