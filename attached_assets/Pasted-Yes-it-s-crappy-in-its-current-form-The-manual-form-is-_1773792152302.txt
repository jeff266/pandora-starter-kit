function generateChartSuggestions(
  sections: ReportSection[],
  skillSummaries: SkillSummary[]
): ChartSuggestion[] {
  const suggestions: ChartSuggestion[] = [];

  // Forecast landing zone — from forecast-rollup
  const forecast = skillSummaries.find(
    s => s.skill_id === 'forecast-rollup'
  );
  if (forecast?.key_metrics) {
    const bear = Number(forecast.key_metrics['bear'])
      || 0;
    const base = Number(forecast.key_metrics['base'])
      || 0;
    const bull = Number(forecast.key_metrics['bull'])
      || 0;
    if (bear > 0 || base > 0 || bull > 0) {
      suggestions.push({
        section_id: 'the_number',
        chart_type: 'bar',
        title: 'Forecast Landing Zone ($K)',
        data: [
          { label: 'Bear',
            value: Math.round(bear / 1000) },
          { label: 'Base',
            value: Math.round(base / 1000) },
          { label: 'Bull',
            value: Math.round(bull / 1000) },
        ],
        rationale: 'Shows forecast range',
      });
    }
  }

  // Rep pipeline comparison — from pipeline-coverage
  // or rep-scorecard
  const coverage = skillSummaries.find(
    s => s.skill_id === 'pipeline-coverage' ||
         s.skill_id === 'rep-scorecard'
  );
  if (coverage?.key_metrics) {
    // Look for rep-level data in key_metrics
    // The summarizer should have stored rep breakdown
    // as key_metrics.rep_1_pipeline, rep_2_pipeline
    // OR as a structured object
    // Check what's actually in key_metrics and extract
    const repData = extractRepData(coverage.key_metrics);
    if (repData.length >= 2) {
      suggestions.push({
        section_id: 'rep_status',
        chart_type: 'horizontal_bar',
        title: 'Open Pipeline by Rep ($K)',
        data: repData,
        rationale: 'Shows rep pipeline concentration',
      });
    }
  }

  // Pipeline movement — from pipeline-waterfall
  const waterfall = skillSummaries.find(
    s => s.skill_id === 'pipeline-waterfall'
  );
  if (waterfall?.key_metrics) {
    const created = Number(
      waterfall.key_metrics['created']) || 0;
    const advanced = Number(
      waterfall.key_metrics['advanced']) || 0;
    const regressed = Number(
      waterfall.key_metrics['regressed']) || 0;
    const won = Number(
      waterfall.key_metrics['closed_won_count']
      || waterfall.key_metrics['won']) || 0;
    const lost = Number(
      waterfall.key_metrics['closed_lost_count']
      || waterfall.key_metrics['lost']) || 0;

    if (created + advanced + regressed + won + lost > 0) {
      suggestions.push({
        section_id: 'the_story',
        chart_type: 'bar',
        title: 'Pipeline Movement This Week',
        data: [
          { label: 'Created', value: created },
          { label: 'Advanced', value: advanced },
          { label: 'Regressed', value: regressed },
          { label: 'Won', value: won },
          { label: 'Lost', value: lost },
        ].filter(d => d.value > 0),
        rationale: 'Shows deal movement',
      });
    }
  }

  return suggestions;
}

function extractRepData(
  metrics: Record<string, string | number>
): { label: string; value: number }[] {
  // Try structured rep data first
  // The coverage summarizer may store this differently
  // Check key names and extract what's there
  const repData: { label: string; value: number }[] = [];

  // Pattern 1: rep_name_pipeline keys
  for (const [key, val] of Object.entries(metrics)) {
    if (key.includes('pipeline') &&
        !key.includes('total') &&
        !key.includes('team')) {
      const label = key
        .replace('_pipeline', '')
        .replace(/_/g, ' ')
        .replace(/\b\w/g, c => c.toUpperCase());
      const value = Math.round(Number(val) / 1000);
      if (value > 0) repData.push({ label, value });
    }
  }

  return repData.slice(0, 6); // max 6 reps
}
