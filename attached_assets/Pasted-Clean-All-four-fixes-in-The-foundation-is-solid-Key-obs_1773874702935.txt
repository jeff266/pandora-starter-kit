## Task 2: Chart trigger detection

After Ask Pandora generates a text response,
check whether a chart should be generated.

Create server/chat/chart-trigger.ts:

```typescript
import { ChartNodeSpec, SkillSummary } from
  '../orchestrator/types';
import { generateChartSpecs } from
  '../orchestrator/chart-intelligence';
import { renderChartFromSpec } from
  '../orchestrator/chart-renderer';

/**
 * Detects whether an Ask Pandora response
 * contains chartable data and if so,
 * generates a chart.
 *
 * Called after the response text is generated.
 * Non-fatal — if chart generation fails,
 * response returns without chart.
 */
export async function maybeGenerateResponseChart(
  question: string,
  responseText: string,
  toolResults: any[],
  workspaceId: string
): Promise<{
  spec: ChartNodeSpec;
  png_base64: string;
  suggested_section_id?: string;
} | null> {

  try {
    // Extract numeric data from tool results
    const chartableData = extractChartableData(
      question, responseText, toolResults
    );

    if (!chartableData) return null;

    // Build a synthetic ReasoningNode-like object
    // for chart intelligence
    const syntheticNode = {
      layer: 'cause' as const,
      question: question,
      answer: responseText.slice(0, 500),
      data_gap: false,
    };

    // Build skill summaries from tool results
    const syntheticSummaries = buildSyntheticSummaries(
      toolResults, chartableData
    );

    // Run chart intelligence (DeepSeek classify)
    const nodes = await generateChartSpecs(
      [syntheticNode],
      syntheticSummaries,
      workspaceId
    );

    const nodeWithSpec = nodes[0];
    if (!nodeWithSpec?.chart_spec?.should_chart)
      return null;

    // Render PNG
    const pngBuffer = await renderChartFromSpec(
      nodeWithSpec.chart_spec
    );

    // Suggest which report section this belongs to
    const suggested_section_id =
      suggestSectionId(question);

    console.log(
      `[AskPandora] Generated chart: ` +
      `"${nodeWithSpec.chart_spec.title}"`
    );

    return {
      spec: nodeWithSpec.chart_spec,
      png_base64: pngBuffer.toString('base64'),
      suggested_section_id,
    };

  } catch (err) {
    console.error(
      '[AskPandora] Chart generation failed:', err
    );
    return null; // non-fatal
  }
}

/**
 * Extracts chartable numeric data from tool
 * call results.
 *
 * Returns null if no numeric data found.
 */
function extractChartableData(
  question: string,
  responseText: string,
  toolResults: any[]
): any | null {

  // Look for deal arrays in tool results
  for (const result of toolResults) {
    const data = result?.content || result?.result
      || result;

    // queryDeals result
    if (Array.isArray(data?.deals) &&
        data.deals.length >= 2) {
      return {
        type: 'deals',
        deals: data.deals.slice(0, 8),
      };
    }
    if (Array.isArray(data) && data.length >= 2 &&
        data[0]?.amount !== undefined) {
      return {
        type: 'deals',
        deals: data.slice(0, 8),
      };
    }

    // Aggregated numeric results
    if (data && typeof data === 'object' &&
        !Array.isArray(data)) {
      const numericEntries = Object.entries(data)
        .filter(([, v]) =>
          typeof v === 'number' && v > 0
        );
      if (numericEntries.length >= 2) {
        return {
          type: 'metrics',
          metrics: Object.fromEntries(numericEntries),
        };
      }
    }
  }

  // Check if response text contains numbers that
  // suggest chartable data (heuristic)
  const dollarAmounts = responseText.match(
    /\$[\d,]+[KMB]?/g
  );
  if (dollarAmounts && dollarAmounts.length >= 3) {
    return { type: 'text_only', amounts: dollarAmounts };
  }

  return null;
}

/**
 * Converts tool results into synthetic SkillSummary
 * objects that chart intelligence can read.
 */
function buildSyntheticSummaries(
  toolResults: any[],
  chartableData: any
): SkillSummary[] {

  if (chartableData.type === 'deals') {
    const deals = chartableData.deals;

    // Build at_risk_deals from deal array
    const atRiskDeals = deals
      .filter((d: any) => d.amount > 0)
      .map((d: any) => ({
        name: d.name || d.deal_name || 'Unknown',
        amount: Number(d.amount) || 0,
        owner: d.owner_email || d.owner || '',
        stage: d.stage_normalized || d.stage || '',
        risk_score: d.deal_risk
          || d.risk_score || 50,
        risk_factors: d.deal_risk_factors || [],
        days_in_stage: d.days_in_stage || 0,
        close_date: d.close_date || '',
        recommended_action: d.recommended_action,
      }));

    return [{
      skill_id: 'ask-pandora-query',
      has_signal: true,
      top_findings: [],
      key_metrics: {
        total_deals: deals.length,
        total_pipeline: deals.reduce(
          (s: number, d: any) =>
            s + Number(d.amount || 0), 0
        ),
      },
      data_age_hours: 0,
      at_risk_deals: atRiskDeals,
      stale_deals: deals
        .filter((d: any) =>
          d.days_since_activity > 14 ||
          d.days_stale > 14
        )
        .map((d: any) => ({
          name: d.name || 'Unknown',
          amount: Number(d.amount) || 0,
          owner: d.owner_email || '',
          stage: d.stage_normalized || '',
          days_stale: d.days_since_activity
            || d.days_stale || 0,
          last_activity_date:
            d.last_activity_date || '',
        })),
    }];
  }

  if (chartableData.type === 'metrics') {
    return [{
      skill_id: 'ask-pandora-query',
      has_signal: true,
      top_findings: [],
      key_metrics: chartableData.metrics,
      data_age_hours: 0,
    }];
  }

  return [];
}

/**
 * Heuristic: which report section does this
 * question relate to?
 */
function suggestSectionId(question: string): string {
  const q = question.toLowerCase();

  if (q.includes('close') || q.includes('forecast')
      || q.includes('quarter') || q.includes('target'))
    return 'node-deal-execution';

  if (q.includes('stage') || q.includes('progress')
      || q.includes('stall') || q.includes('stuck')
      || q.includes('pipeline'))
    return 'node-pipeline-conv';

  if (q.includes('rep') || q.includes('nate')
      || q.includes('sara') || q.includes('team')
      || q.includes('attainment'))
    return 'node-team-execution';

  if (q.includes('q2') || q.includes('next quarter')
      || q.includes('coverage') || q.includes('future'))
    return 'node-forward-look';

  return 'node-deal-execution'; // default
}
```

## Task 4: Current report context endpoint

Add to server/routes/agents.ts:

```typescript
// GET /:workspaceId/reports/current
// Returns the most recent report document for
// this workspace — used by "Add to report" button
// to know which report_document_id to target

router.get(
  '/:workspaceId/reports/current',
  async (req, res) => {
    const { workspaceId } = req.params;
    // Verify workspace access

    const result = await query(`
      SELECT id, document_type, week_label,
        generated_at,
        jsonb_array_length(sections) as section_count,
        sections->0->>'id' as first_section_id
      FROM report_documents
      WHERE workspace_id = $1
      ORDER BY generated_at DESC
      LIMIT 1
    `, [workspaceId]);

    if (!result.rows[0]) {
      return res.status(404).json(
        { error: 'No reports found' }
      );
    }

    res.json(result.rows[0]);
  }
);
```
