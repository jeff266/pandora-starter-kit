**Replit Prompt: Command Palette (Cmd+K)**

Read these files first:
1. `App.tsx` or the root layout component — find where global keyboard shortcuts are registered
2. The sidebar navigation component — find all route definitions (these become page results)
3. How deals and reps are queried — find the existing API endpoints for deal lists and rep lists
4. The `openAskPandora` utility — you'll call this for deal results
5. The skill registry endpoint — `GET /api/workspaces/:id/skills`

---

### What to build

A command palette triggered by Cmd+K (Mac) / Ctrl+K (Windows). Fuzzy search across 5 categories. Pure client-side matching after an initial data fetch. No LLM, no tokens.

---

### Part 1 — Data fetch on workspace load

When the workspace loads, fetch and cache the searchable index in memory. This runs once, not on every keystroke.

```typescript
// server: add one endpoint
GET /api/workspaces/:id/search-index

// Returns:
{
  deals: [
    {
      id: string,
      name: string,           // expanded — run expandDealName()
      amount: number,
      stage: string,
      ownerName: string,
      daysSinceActivity: number,
      closeDate: string,
    }
  ],  // all open deals, max 500
  reps: [
    { email: string, name: string, role: string }
  ],
  hypotheses: [
    { id: string, metric: string, hypothesis: string, status: string }
  ],
  skills: [
    { id: string, name: string, category: string }
  ]
}
```

Static pages and skills are hardcoded on the client — they don't need an API call.

Cache the response in React state or a module-level variable. Refresh every 5 minutes or on manual sync.

**Result builder** — runs on every keystroke, debounced 80ms:

```typescript
function buildResults(query: string): SearchResult[] {
  if (!query.trim()) return getDefaultResults()  // show recent + top actions when empty

  const results: SearchResult[] = []

  // Deals — show top 4
  dealFuse.search(query).slice(0, 4).forEach(({ item }) => {
    results.push({
      type: 'deal',
      id: item.id,
      title: item.name,
      subtitle: `${item.stage} · $${formatCurrency(item.amount)} · ${item.ownerName} · ${item.daysSinceActivity}d silent`,
      icon: '📋',
      action: () => {
        openAskPandora({ type: 'deal', entity_id: item.id, entity_name: item.name })
        close()
      },
    })
  })

  // Reps — show top 2
  repFuse.search(query).slice(0, 2).forEach(({ item }) => {
    results.push({
      type: 'rep',
      id: item.email,
      title: item.name,
      subtitle: `Rep · ${item.email}`,
      icon: '👤',
      action: () => {
        navigate(`/reps/${encodeURIComponent(item.email)}`)
        close()
      },
    })
  })

  // Pages — show top 3
  const pageQuery = query.toLowerCase()
  PAGES.filter(p =>
    p.title.toLowerCase().includes(pageQuery) ||
    p.subtitle.toLowerCase().includes(pageQuery)
  ).slice(0, 3).forEach(page => {
    results.push({
      type: 'page',
      id: page.id,
      title: page.title,
      subtitle: page.subtitle,
      icon: '→',
      action: () => { navigate(page.route); close() },
    })
  })

  // Hypotheses — show top 2
  hypothesisFuse.search(query).slice(0, 2).forEach(({ item }) => {
    results.push({
      type: 'hypothesis',
      id: item.id,
      title: item.hypothesis.slice(0, 60) + (item.hypothesis.length > 60 ? '...' : ''),
      subtitle: `Hypothesis · ${item.metric} · ${item.status}`,
      icon: '📊',
      action: () => {
        navigate(`/actions?tab=hypotheses&highlight=${item.id}`)
        close()
      },
    })
  })

  // Skills — show top 2
  const skillMatches = searchIndex.skills.filter(s =>
    s.name.toLowerCase().includes(pageQuery)
  ).slice(0, 2)
  skillMatches.forEach(skill => {
    results.push({
      type: 'skill',
      id: skill.id,
      title: `Run: ${skill.name}`,
      subtitle: `Skill · ${skill.category}`,
      icon: '⚡',
      action: () => {
        triggerSkillRun(skill.id)
        close()
      },
    })
  })

  // Always offer "Ask Pandora" as last result if query is a question
  if (query.length > 10) {
    results.push({
      type: 'ask',
      id: 'ask-pandora',
      title: `Ask Pandora: "${query.slice(0, 40)}${query.length > 40 ? '...' : ''}"`,
      subtitle: 'Open in Ask Pandora',
      icon: '✦',
      action: () => {
        openAskPandora({ prefillQuery: query })
        close()
      },
    })
  }

  return results
}

function getDefaultResults(): SearchResult[] {
  // When palette opens with no query:
  // Show: top 3 deals by days silent, top 3 pages, "Ask Pandora" prompt
  const stalestDeals = [...searchIndex.deals]
    .sort((a, b) => b.daysSinceActivity - a.daysSinceActivity)
    .slice(0, 3)

  return [
    ...stalestDeals.map(deal => ({
      type: 'deal' as const,
      id: deal.id,
      title: deal.name,
      subtitle: `${deal.daysSinceActivity}d silent · $${formatCurrency(deal.amount)} · ${deal.ownerName}`,
      icon: '📋',
      action: () => {
        openAskPandora({ type: 'deal', entity_id: deal.id, entity_name: deal.name })
        close()
      },
    })),
    { type: 'page', id: 'actions', title: 'Actions', subtitle: 'Sprint view', icon: '→',
      action: () => { navigate('/actions'); close() } },
    { type: 'page', id: 'concierge', title: 'Concierge', subtitle: 'Weekly briefing', icon: '→',
      action: () => { navigate('/concierge'); close() } },
    { type: 'ask', id: 'ask', title: 'Ask Pandora anything...', subtitle: 'Open Ask Pandora',
      icon: '✦', action: () => { openAskPandora({}); close() } },
  ]
}
```

### Part 4 — Keyboard handler

```typescript
function handleKeyDown(e: React.KeyboardEvent) {
  switch (e.key) {
    case 'ArrowDown':
      e.preventDefault()
      setSelectedIndex(i => Math.min(i + 1, results.length - 1))
      break
    case 'ArrowUp':
      e.preventDefault()
      setSelectedIndex(i => Math.max(i - 1, 0))
      break
    case 'Enter':
      e.preventDefault()
      if (results[selectedIndex]) results[selectedIndex].action()
      break
    case 'Escape':
      close()
      break
  }
}
```

### Part 5 — Global registration

In `App.tsx` or the root layout:

```typescript
useEffect(() => {
  const handler = (e: KeyboardEvent) => {
    if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
      e.preventDefault()
      setCommandPaletteOpen(true)
    }
  }
  window.addEventListener('keydown', handler)
  return () => window.removeEventListener('keydown', handler)
}, [])
```
