### New compute function

Create `server/analysis/behavioral-stage-correction.ts`:

```typescript
import { query } from '../db'

export interface BehavioralCorrection {
  dealId: string
  dealName: string
  dealAmount: number
  ownerEmail: string

  // CRM state
  crmStage: string
  crmStageNormalized: string
  crmCloseProb: number
  crmExpectedValue: number       // amount × crmCloseProb

  // Behavioral state (from stage-divergence skill output)
  behavioralStage: string
  behavioralStageNormalized: string
  behavioralCloseProb: number
  behavioralExpectedValue: number  // amount × behavioralCloseProb

  // The correction
  evDelta: number                // behavioral EV - CRM EV
  direction: 'understated' | 'overstated'  // is CRM understating or overstating?
  divergenceSignals: string[]    // what drove the divergence
  divergenceConfidence: number   // 0–1 from stage-divergence output
}

export interface BehavioralAdjustedEV {
  // The 6th bearing
  bearingValue: number           // sum of behavioral EVs for all corrected deals
                                 // + CRM stage EVs for uncorrected deals

  // What changed vs CRM-stage EV
  totalEvDelta: number           // net adjustment (can be positive or negative)
  corrections: BehavioralCorrection[]

  // Summary counts
  understatedDeals: number       // CRM is behind behavioral reality
  overstatedDeals: number        // CRM is ahead of behavioral reality
  correctedDealCount: number
  uncorrectedDealCount: number   // deals with no divergence signal

  // Data quality
  divergenceSkillAge: number     // how many hours since stage-divergence last ran
  isStale: boolean               // true if > 48 hours since last divergence run
}

export async function computeBehavioralAdjustedEV(
  workspaceId: string,
  openDeals: { id: string; name: string; amount: number; stageNormalized: string; ownerEmail: string }[],
  stageCloseProbabilities: Record<string, number>
): Promise<BehavioralAdjustedEV> {

  // Step 1: Load most recent stage-divergence skill run
  const divergenceRun = await query(`
    SELECT result_data, started_at
    FROM skill_runs
    WHERE workspace_id = $1
      AND skill_id = 'stage-divergence'
      AND status = 'completed'
    ORDER BY started_at DESC
    LIMIT 1
  `, [workspaceId])

  if (divergenceRun.rows.length === 0) {
    // No divergence data — return CRM-stage EV as fallback
    const crmStageEV = openDeals.reduce((sum, deal) => {
      const prob = stageCloseProbabilities[deal.stageNormalized] ?? 0.25
      return sum + deal.amount * prob
    }, 0)

    return {
      bearingValue: crmStageEV,
      totalEvDelta: 0,
      corrections: [],
      understatedDeals: 0,
      overstatedDeals: 0,
      correctedDealCount: 0,
      uncorrectedDealCount: openDeals.length,
      divergenceSkillAge: Infinity,
      isStale: true,
    }
  }

  const divergenceResult = divergenceRun.rows[0].result_data
  const runAgeHours = (Date.now() - new Date(divergenceRun.rows[0].started_at).getTime()) / 3600000

  // Step 2: Extract divergent deals from skill output
  // Stage-divergence skill should output a list of deals with divergence signals
  // Find the correct path in the result_data — likely result_data.divergentDeals or similar
  const divergentDeals: {
    dealId: string
    dealName: string
    crmStage: string
    behavioralStage: string
    signals: string[]
    confidence: number
  }[] = divergenceResult?.divergentDeals
    ?? divergenceResult?.findings?.filter((f: any) => f.type === 'stage_divergence')
       .map((f: any) => ({
         dealId: f.dealId,
         dealName: f.dealName,
         crmStage: f.crmStage,
         behavioralStage: f.behavioralStage,
         signals: f.signals ?? [],
         confidence: f.confidence ?? 0.7,
       }))
    ?? []

  // Build a lookup map for fast access
  const divergenceMap = new Map(divergentDeals.map(d => [d.dealId, d]))

  // Step 3: Compute corrected EV for each deal
  const corrections: BehavioralCorrection[] = []
  let bearingValue = 0
  let totalEvDelta = 0

  for (const deal of openDeals) {
    const crmProb = stageCloseProbabilities[deal.stageNormalized] ?? 0.25
    const crmEV = deal.amount * crmProb

    const divergence = divergenceMap.get(deal.id)

    if (divergence && divergence.confidence >= 0.6) {
      // Use behavioral stage probability
      const behavioralStageNorm = normalizeStage(divergence.behavioralStage)
      const behavioralProb = stageCloseProbabilities[behavioralStageNorm] ?? crmProb
      const behavioralEV = deal.amount * behavioralProb
      const evDelta = behavioralEV - crmEV

      corrections.push({
        dealId: deal.id,
        dealName: deal.name,
        dealAmount: deal.amount,
        ownerEmail: deal.ownerEmail,
        crmStage: divergence.crmStage,
        crmStageNormalized: deal.stageNormalized,
        crmCloseProb: crmProb,
        crmExpectedValue: crmEV,
        behavioralStage: divergence.behavioralStage,
        behavioralStageNormalized: behavioralStageNorm,
        behavioralCloseProb: behavioralProb,
        behavioralExpectedValue: behavioralEV,
        evDelta,
        direction: evDelta > 0 ? 'understated' : 'overstated',
        divergenceSignals: divergence.signals,
        divergenceConfidence: divergence.confidence,
      })

      bearingValue += behavioralEV
      totalEvDelta += evDelta
    } else {
      // No divergence — use CRM stage as-is
      bearingValue += crmEV
    }
  }

  const correctedIds = new Set(corrections.map(c => c.dealId))

  return {
    bearingValue,
    totalEvDelta,
    corrections,
    understatedDeals: corrections.filter(c => c.direction === 'understated').length,
    overstatedDeals: corrections.filter(c => c.direction === 'overstated').length,
    correctedDealCount: corrections.length,
    uncorrectedDealCount: openDeals.filter(d => !correctedIds.has(d.id)).length,
    divergenceSkillAge: runAgeHours,
    isStale: runAgeHours > 48,
  }
}

// Normalize a stage string to match the stageCloseProbabilities keys
// Reuse the existing normalization function from the codebase
function normalizeStage(stage: string): string {
  // Find the existing normalizeStage function in the codebase and call it
  // It's likely in server/connectors/hubspot/transforms.ts or similar
  // For now, basic normalization:
  return stage.toLowerCase().replace(/[^a-z0-9]/g, '_')
}
```
