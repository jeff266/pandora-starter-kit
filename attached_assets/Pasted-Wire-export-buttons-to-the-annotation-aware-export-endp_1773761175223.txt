async function handleExport(format: 'pdf' | 'docx' | 'pptx') {
  if (!reportDocument?.id || !workspaceId) return;

  try {
    const res = await fetch(
      `/api/workspaces/${workspaceId}/reports/` +
      `${reportDocument.id}/export`,
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${token}`,
        },
        body: JSON.stringify({ format }),
      }
    );

    if (!res.ok) {
      console.error('Export failed:', await res.text());
      return;
    }

    const data = await res.json();
    console.log('[Export] Merged document:', data);
    // Phase 5 will handle actual file download here
    // For now this confirms annotations are merged correctly

  } catch (err) {
    console.error('[Export] Error:', err);
  }
}

If the export endpoint doesn't exist in agents.ts, add it:

router.post(
  '/:workspaceId/reports/:reportId/export',
  async (req, res) => {
    const { workspaceId, reportId } = req.params;
    const { format = 'pdf' } = req.body;

    // Verify ownership
    const doc = await getReportDocumentById(reportId, workspaceId);
    if (!doc) return res.status(404).json({ error: 'Not found' });

    // Merge annotations
    const { mergeAnnotationsForExport } = await import(
      '../orchestrator/annotation-merge'
    );
    const merged = await mergeAnnotationsForExport(doc, workspaceId);

    res.json({
      report_document: merged,
      format,
      annotations_applied: 'merged',
    });
  }
);
