### Part 1 — Group findings by pipeline type in the opening brief

In `server/briefing/opening-brief.ts`, extend the big deals at risk query to include pipeline/deal type grouping:

```typescript
// After fetching bigDealsAtRisk, group them by pipeline type
// Pipeline type priority order for display:
// 1. renewal (highest urgency — revenue at risk)
// 2. expansion (existing customers)
// 3. new_business (net new)
// 4. other / unknown

export type DealGroup = 'renewal' | 'expansion' | 'new_business' | 'other'

export interface GroupedDealFindings {
  group: DealGroup
  label: string          // "Renewal", "Expansion", "New Business"
  deals: BigDealAtRisk[]
  totalValue: number
  criticalCount: number
}

function classifyDealGroup(deal: BigDealAtRisk, workspaceConfig: WorkspaceConfig): DealGroup {
  // 1. Check deal_type field directly if it exists
  if (deal.deal_type) {
    const t = deal.deal_type.toLowerCase()
    if (t.includes('renewal') || t.includes('renew')) return 'renewal'
    if (t.includes('expansion') || t.includes('expand') || t.includes('upsell')) return 'expansion'
    if (t.includes('new') || t.includes('net new')) return 'new_business'
  }

  // 2. Check pipeline name from workspace config
  if (deal.pipelineId && workspaceConfig.pipelines) {
    const pipeline = workspaceConfig.pipelines.find(p => p.id === deal.pipelineId)
    if (pipeline) {
      const name = pipeline.name.toLowerCase()
      if (name.includes('renewal')) return 'renewal'
      if (name.includes('expansion')) return 'expansion'
      if (name.includes('new') || name.includes('core')) return 'new_business'
    }
  }

  // 3. Check deal name patterns as fallback
  const name = deal.name.toLowerCase()
  if (name.includes('renewal') || name.includes('renew')) return 'renewal'
  if (name.includes('expansion') || name.includes('expand')) return 'expansion'

  return 'other'
}

export function groupDealFindings(
  deals: BigDealAtRisk[],
  workspaceConfig: WorkspaceConfig
): GroupedDealFindings[] {
  const groupOrder: DealGroup[] = ['renewal', 'expansion', 'new_business', 'other']
  const groupLabels: Record<DealGroup, string> = {
    renewal: 'Renewal',
    expansion: 'Expansion',
    new_business: 'New Business',
    other: 'Pipeline',
  }

  const groups = new Map<DealGroup, BigDealAtRisk[]>()

  for (const deal of deals) {
    const group = classifyDealGroup(deal, workspaceConfig)
    if (!groups.has(group)) groups.set(group, [])
    groups.get(group)!.push(deal)
  }

  // Only return groups that have deals, in priority order
  return groupOrder
    .filter(g => groups.has(g))
    .map(g => ({
      group: g,
      label: groupLabels[g],
      deals: groups.get(g)!,
      totalValue: groups.get(g)!.reduce((sum, d) => sum + (d.amount ?? 0), 0),
      criticalCount: groups.get(g)!.filter(d => d.severity === 'critical').length,
    }))
}
```

In the opening brief assembly, detect whether grouping is meaningful:

### Part 3 — Custom view filters via All Data dropdown

// types/concierge-filters.ts

export interface ConciergeFilter {
  id: string
  label: string
  type: 'built_in' | 'custom'
  filters: {
    repEmail?: string          // filter to specific rep
    pipelineGroup?: DealGroup  // 'renewal' | 'expansion' | 'new_business'
    pipelineId?: string        // specific HubSpot/SF pipeline
    region?: string            // territory/region field value
    minAmount?: number
  }
}

// Built-in filters — always present
const BUILT_IN_FILTERS: ConciergeFilter[] = [
  {
    id: 'all',

    type: 'built_in',
    filters: {}
  },
  {
    id: 'renewal',
    label: 'Renewal Only',
    type: 'built_in',
    filters: { pipelineGroup: 'renewal' }

    id: 'expansion',
    label: 'Expansion Only',
    type: 'built_in',
