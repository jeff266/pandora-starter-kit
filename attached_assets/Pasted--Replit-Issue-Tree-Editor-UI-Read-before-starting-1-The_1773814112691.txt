## Task 1: Issue Tree API endpoints

Add to server/routes/agents.ts:

```
GET /:workspaceId/agents/:agentId/issue-tree
  Returns: IssueTreeNode[] ordered by position
  Returns [] if no nodes exist

POST /:workspaceId/agents/:agentId/issue-tree
  Body: { node_id, title, standing_question,
          mece_category, primary_skill_ids[], position }
  Upserts on (agent_id, node_id)
  Returns: saved node

PATCH /:workspaceId/agents/:agentId/issue-tree/:nodeId
  Body: partial node fields
  Returns: updated node

DELETE /:workspaceId/agents/:agentId/issue-tree/:nodeId
  Returns: { deleted: true }

PATCH /:workspaceId/agents/:agentId/issue-tree/reorder
  Body: { positions: [{ node_id, position }] }
  Batch updates position for all nodes in one call
  Returns: { updated: N }

PATCH /:workspaceId/agents/:agentId
  Already exists — ensure it handles:
  use_issue_tree: boolean
  custom_skill_ids: string[] | null
```

All endpoints enforce workspace ownership —
verify agent belongs to workspace before read/write.

function generateNodeId(title: string): string {
  return title
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '_')
    .replace(/^_|_$/g, '')
    .substring(0, 40);
}

### Drag to reorder

Use HTML5 drag and drop — no library needed.

```typescript
function handleDragStart(e: React.DragEvent,
                          index: number) {
  e.dataTransfer.setData('text/plain',
    String(index));
  e.dataTransfer.effectAllowed = 'move';
}

function handleDragOver(e: React.DragEvent,
                         index: number) {
  e.preventDefault();
  e.dataTransfer.dropEffect = 'move';
  setDragOverIndex(index);
}

function handleDrop(e: React.DragEvent,
                     targetIndex: number) {
  e.preventDefault();
  const sourceIndex = parseInt(
    e.dataTransfer.getData('text/plain')
  );
  if (sourceIndex === targetIndex) return;

  const reordered = [...nodes];
  const [moved] = reordered.splice(sourceIndex, 1);
  reordered.splice(targetIndex, 0, moved);

  // Update positions
  const updated = reordered.map((n, i) => ({
    ...n, position: i + 1
  }));
  setNodes(updated);

  // Persist reorder
  fetch(
    `/api/workspaces/${workspaceId}/agents/
      ${agentId}/issue-tree/reorder`,
    {
      method: 'PATCH',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${token}`,
      },
      body: JSON.stringify({
        positions: updated.map(n => ({
          node_id: n.node_id,
          position: n.position,
        })),
      }),
    }
  );

  setDragOverIndex(null);
}
```

Show a visual drop indicator (blue horizontal line)
at dragOverIndex position between cards.

## Task 3: MECECoverageIndicator Component

Small inline component that shows which MECE
categories are covered vs. missing.

```typescript
const MECE_CATEGORIES = [
  { id: 'generation', label: 'Generation',
    description: 'Top of funnel, pipeline creation' },
  { id: 'conversion', label: 'Conversion',
    description: 'Mid funnel, stage progression' },
  { id: 'execution',  label: 'Execution',
    description: 'Late funnel, closing and forecast' },
  { id: 'retention',  label: 'Retention',
    description: 'Post-close, expansion and churn' },
];

function MECECoverageIndicator({
  nodes
}: { nodes: IssueTreeNode[] }) {
  const covered = new Set(
    nodes.map(n => n.mece_category)
  );

  return (
    <div style={{
      display: 'flex', gap: 6, marginBottom: 16,
      flexWrap: 'wrap',
    }}>
      {MECE_CATEGORIES.map(cat => {
        const isCovered = covered.has(cat.id);
        return (
          <div
            key={cat.id}
            title={cat.description}
            style={{
              padding: '3px 10px',
              borderRadius: 4,
              fontSize: 11,
              fontWeight: 500,
              background: isCovered
                ? '#F0FDF9' : '#F8FAFC',
              color: isCovered
                ? '#0D9488' : '#94A3B8',
              border: '0.5px solid',
              borderColor: isCovered
                ? '#0D9488' : '#E2E8F0',
            }}
          >
            {isCovered ? '✓' : '○'} {cat.label}
          </div>
        );
      })}
      {nodes.some(n => n.mece_category === 'custom') && (
        <div style={{
          padding: '3px 10px',
          borderRadius: 4,
          fontSize: 11,
          fontWeight: 500,
          background: '#F0FDF9',
          color: '#0D9488',
          border: '0.5px solid #0D9488',
        }}>
          ✓ Custom
        </div>
      )}
    </div>
  );
}
```
