> **Part 1 — Database: tag actions to hypotheses**
>
> Add to the actions table:
> ```sql
> ALTER TABLE actions
>   ADD COLUMN IF NOT EXISTS hypothesis_id UUID REFERENCES standing_hypotheses(id),
>   ADD COLUMN IF NOT EXISTS sprint_week DATE,  -- which week's sprint this belongs to
>   ADD COLUMN IF NOT EXISTS expected_value_delta NUMERIC,  -- $ impact on P50
>   ADD COLUMN IF NOT EXISTS effort TEXT CHECK (effort IN ('immediate', 'this_week', 'this_month')),
>   ADD COLUMN IF NOT EXISTS state TEXT NOT NULL DEFAULT 'pending'
>     CHECK (state IN ('pending', 'in_progress', 'executed', 'deferred', 'not_applicable', 'escalated'));
> ```
>
> Replace the existing `status` field logic with the new `state` values:
> - `pending` → needs a decision
> - `in_progress` → user said "working on it" — crumb trail fires
> - `executed` → done, written to CRM
> - `deferred` → right action, move to next week's sprint
> - `not_applicable` → wrong context, feeds training data
> - `escalated` → needs leadership, routes to Slack
>
> Add a weekly sprint assembly job — runs Monday at 8:30 AM after all skills complete:
>
> ```typescript
> // server/jobs/assemble-weekly-sprint.ts
>
> export async function assembleWeeklySprint(workspaceId: string): Promise<void> {
>   const thisWeek = startOfWeek(new Date())
>
>   // 1. Load all-else-equal action menu from latest monte-carlo run
>   const monteCarloRun = await getLatestSkillRun(workspaceId, 'monte-carlo-forecast')
>   const actionMenu = monteCarloRun?.result_data?.allElseEqual?.actionMenu ?? []
>
>   // 2. Load active standing hypotheses
>   const hypotheses = await query(`
>     SELECT * FROM standing_hypotheses
>     WHERE workspace_id = $1 AND status = 'active'
>     ORDER BY
>       CASE WHEN current_value < alert_threshold THEN 0 ELSE 1 END,
>       created_at ASC
>   `, [workspaceId])
>
>   // 3. For each action in the menu, find its parent hypothesis
>   //    Match by metric name or action type
>   for (const action of actionMenu.slice(0, 7)) {
>     const parentHypothesis = findParentHypothesis(action, hypotheses.rows)
>
>     // Upsert into actions table with sprint_week and hypothesis_id
>     await query(`
>       INSERT INTO actions (
>         workspace_id, title, action_type, severity,
>         expected_value_delta, effort, sprint_week,
>         hypothesis_id, state, source
>       ) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, 'pending', 'monte_carlo')
>       ON CONFLICT (workspace_id, title, sprint_week)
>       DO UPDATE SET
>         expected_value_delta = EXCLUDED.expected_value_delta,
>         hypothesis_id = EXCLUDED.hypothesis_id
>     `, [
>       workspaceId,
>       action.label,
>       action.actionType,
>       action.rank <= 2 ? 'critical' : action.rank <= 5 ? 'warning' : 'notable',
>       action.expectedValueIfDone,
>       action.effort,
>       thisWeek,
>       parentHypothesis?.id ?? null,
>     ])
>   }
>
>   // 4. Also promote any standing hypothesis threshold breaches
>   //    as sprint actions if not already present
>   for (const hyp of hypotheses.rows) {
>     if (isThresholdBreached(hyp)) {
>       await upsertHypothesisAlert(workspaceId, hyp, thisWeek)
>     }
>   }
> }
>
> function findParentHypothesis(action: ActionMenuItem, hypotheses: any[]): any | null {
>   // Match action type to hypothesis metric
>   const matchMap: Record<string, string[]> = {
>     're_engage_deal': ['large_deal_win_rate', 'large_deal_cohort'],
>     'close_deal': ['large_deal_win_rate', 'conversion_rate'],
>     'generate_pipeline': ['pipeline_coverage_ratio'],
>     'improve_lever': ['conversion_rate', 'smb_sales_cycle_days'],
>   }
>   const metrics = matchMap[action.actionType] ?? []
>   return hypotheses.find(h => metrics.includes(h.metric)) ?? null
> }
> ```

> **Part 2 — API: sprint and hypothesis endpoints**
>
> Add to the actions routes:
>
> ```
> GET /api/workspaces/:id/actions/sprint
>   Returns: { week, actions: (top 7 tagged to hypotheses), hypotheses, summary }
>
> GET /api/workspaces/:id/actions/backlog
>   Returns: { actions: (all non-sprint actions, paginated), total }
>
> PATCH /api/workspaces/:id/actions/:actionId/state
>   Body: { state: 'in_progress' | 'executed' | 'deferred' | 'not_applicable' | 'escalated' }
>   Side effects:
>     - 'in_progress': fires recordCrumbTrail() with signal = 'working_on_it'
>     - 'executed': creates CRM task if action has a dealId
>     - 'deferred': sets sprint_week = next Monday
>     - 'escalated': sends Slack DM to workspace admin
> ```
