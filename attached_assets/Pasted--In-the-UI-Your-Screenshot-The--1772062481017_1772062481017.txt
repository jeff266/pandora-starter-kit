  -- To query all Ask Pandora history for a    
  specific deal:                               
  SELECT                                       
    csm.role,                                  
    csm.content,                               
    csm.metadata,                              
    csm.created_at                             
  FROM chat_session_messages csm               
  JOIN chat_sessions cs ON cs.id =             
  csm.session_id                               
  WHERE cs.workspace_id = '<workspace_id>'     
    AND cs.entity_type = 'deal'                
    AND cs.entity_id = '<deal_id>'             
  ORDER BY csm.created_at ASC;
