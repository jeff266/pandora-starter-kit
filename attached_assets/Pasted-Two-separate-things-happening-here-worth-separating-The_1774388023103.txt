function buildStageMappingTablePrompt(
  unmappedStages: UnmappedStage[],
  knownFunnelPositions: string[]
): string {
  const rows = unmappedStages
    .map(s => {
      const guess = inferStageMapping(s.name)

      const value = formatCurrency(s.total_value)
