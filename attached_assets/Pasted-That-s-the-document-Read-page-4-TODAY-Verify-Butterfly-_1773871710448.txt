function applyOrientationRule(
  suggestedType: string,
  dataPoints: ChartDataPoint[]
): string {
  // Never override donut or line
  if (suggestedType === 'donut' ||
      suggestedType === 'doughnut' ||
      suggestedType === 'line') {
    return suggestedType;
  }

  // Unconditional: any label > 12 chars → horizontal
  const hasLongLabel = dataPoints.some(
    d => d.label.length > 12
  );
  if (hasLongLabel) return 'horizontalBar';

  return suggestedType || 'bar';
