## What to build

Each report section gets a chart area below its content.
The chart area has three states:

STATE 1 — No chart accepted yet (default)
Shows the AI suggestion as a preview card:

┌─────────────────────────────────────────┐
│  Suggested chart                    [×] │
│  "Forecast Landing Zone" · bar chart    │
│  ┌─────────────────────────────────┐    │
│  │   [chart preview image]         │    │
│  └─────────────────────────────────┘    │
│  [ Change type ▾ ]    [ Add to report ] │
└─────────────────────────────────────────┘

STATE 2 — Chart accepted (user clicked "Add to report")
Shows the chart inline with section content:

┌─────────────────────────────────────────┐
│  [chart image — full width]             │
│  "Forecast Landing Zone"                │
│  [ Edit ]  [ Remove ]                   │
└─────────────────────────────────────────┘

STATE 3 — No suggestion, no chart
Shows a subtle "Add chart" link:
  + Add chart to this section

State:
```typescript

  useState<ChartSuggestion | null>(null);
const [acceptedChart, setAcceptedChart] =
  useState<ReportChart | null>(null);
const [previewUrl, setPreviewUrl] =
  useState<string | null>(null);
const [loading, setLoading] = useState(true);

On mount, load both suggestion and accepted chart:
```typescript
useEffect(() => {
  async function load() {
    setLoading(true);
    try {
      // Load AI suggestion for this section
      const sugRes = await fetch(
        `/api/workspaces/${workspaceId}/reports/` +
        `${reportDocumentId}/chart-suggestions`,
        { headers: { Authorization: `Bearer ${token}` } }
      );
      const suggestions = await sugRes.json();
      const match = suggestions.find(
        (s: any) => s.section_id === sectionId
      );
      if (match) {
        setSuggestion(match);
        setSelectedType(match.chart_type);
      }

      // Load accepted chart for this section
      const chartRes = await fetch(
        `/api/workspaces/${workspaceId}/reports/` +
        `${reportDocumentId}/charts`,
        { headers: { Authorization: `Bearer ${token}` } }
      );
      const charts = await chartRes.json();
      const accepted = charts.find(
        (c: any) => c.section_id === sectionId
      );
      if (accepted) {
        setAcceptedChart(accepted);
        // Load the PNG for display
        const imgRes = await fetch(
          `/api/workspaces/${workspaceId}/reports/` +
          `${reportDocumentId}/charts/${accepted.id}/image`,
          { headers: { Authorization: `Bearer ${token}` } }
        );
        const blob = await imgRes.blob();
        setPreviewUrl(URL.createObjectURL(blob));
      }
    } finally {
      setLoading(false);
    }
  }
  load();
}, [reportDocumentId, sectionId]);
```

### Add chart to report:
```typescript
async function handleAddToReport() {
  if (!suggestion) return;
  setAdding(true);
  try {
    const res = await fetch(
      `/api/workspaces/${workspaceId}/reports/` +
      `${reportDocumentId}/charts`,
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${token}`,
        },
        body: JSON.stringify({
          section_id: sectionId,
          chart_type: selectedType || suggestion.chart_type,
          title: suggestion.title,
          data: suggestion.data,
          x_label: suggestion.x_label,
          y_label: suggestion.y_label,
        }),
      }
    );
    const saved = await res.json();
    setAcceptedChart(saved);

    // Load the PNG
    const imgRes = await fetch(
      `/api/workspaces/${workspaceId}/reports/` +
      `${reportDocumentId}/charts/${saved.id}/image`,
      { headers: { Authorization: `Bearer ${token}` } }
    );
    const blob = await imgRes.blob();
    setPreviewUrl(URL.createObjectURL(blob));

  } finally {
    setAdding(false);
  }
}
```

### Remove chart:
```typescript
async function handleRemove() {
  if (!acceptedChart) return;
  await fetch(
    `/api/workspaces/${workspaceId}/reports/` +
    `${reportDocumentId}/charts/${acceptedChart.id}`,
    {
      method: 'DELETE',
      headers: { Authorization: `Bearer ${token}` },
    }
  );
  setAcceptedChart(null);
  if (previewUrl) URL.revokeObjectURL(previewUrl);
  setPreviewUrl(null);
}
```

## Task 3: Image endpoint

Check whether
GET /:workspaceId/reports/:reportId/charts/:chartId/image
already exists in server/routes/agents.ts.

If it exists and returns the PNG buffer with
Content-Type: image/png — no changes needed.

If it doesn't exist, add it:
```typescript
router.get(
  '/:workspaceId/reports/:reportId/charts/:chartId/image',
  async (req, res) => {
    const { workspaceId, reportId, chartId } = req.params;

    const result = await query(`
      SELECT rc.chart_png, rc.chart_type
      FROM report_charts rc
      JOIN report_documents rd
        ON rd.id = rc.report_document_id
      WHERE rc.id = $1
        AND rc.report_document_id = $2
        AND rd.workspace_id = $3
    `, [chartId, reportId, workspaceId]);

    if (!result.rows[0]?.chart_png) {
      return res.status(404).json(
        { error: 'Chart not found' }
      );
    }

    res.setHeader('Content-Type', 'image/png');
    res.setHeader('Cache-Control', 'public, max-age=3600');
    res.end(result.rows[0].chart_png);
  }
);
```
