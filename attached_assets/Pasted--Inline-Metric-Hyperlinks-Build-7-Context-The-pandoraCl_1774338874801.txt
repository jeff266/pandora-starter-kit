## Change 1 — claim-tag-processor.ts

File: server/reports/claim-tag-processor.ts

In the function that builds pandoraClaim mark attrs,
add numeric detection before the mark is constructed.
```typescript
const NUMERIC_PATTERNS = [
  /^\$[\d,.]+[MKBmkb]?$/,          // $111M, $6.1M, $800K
  /^\d[\d,]*\s+deals?$/i,           // 162 deals, 1 deal
  /^\d[\d,]*\s+accounts?$/i,        // 14 accounts
  /^\d[\d,]*\s+reps?$/i,            // 5 reps
  /^\d[\d,]*\s+contacts?$/i,        // 23 contacts
  /^[\d.]+%$/,                       // 34%, 100%
  /^[\d.]+x$/,                       // 111x, 2.2x
  /^\d+\s+days?$/i,                  // 45 days
  /^\d[\d,]*$/,                      // standalone number
]

function isNumericText(text: string): boolean {
  const trimmed = text.trim()
  return NUMERIC_PATTERNS.some(p => p.test(trimmed))
}
```

When building the mark attrs object, add the field:
```typescript
// In the section that constructs mark attrs:
marks: [{
  type: 'pandoraClaim',
  attrs: {
    claim_id:     ann.claim_id,
    skill_id:     ann.skill_id,
    skill_run_id: skillRunId,
    metric_name:  ann.metric_name,
    severity:     ann.severity,
    data_numeric: isNumericText(ann.text) ? 'true' : 'false',
  }
}]
```

If data_numeric is already being set in a previous build,
verify the isNumericText() logic matches the patterns
above and update if needed.

```typescript
// State needed (add if not present):
const [claimDrawerOpen, setClaimDrawerOpen] =
  useState(false)
const [claimDrawerLoading, setClaimDrawerLoading] =
  useState(false)
const [claimDrawerData, setClaimDrawerData] =
  useState<any>(null)
const [claimDrawerError, setClaimDrawerError] =
  useState<string | null>(null)

// Function:
async function handleQuestionClaim(claim: {
  claim_id: string
  skill_run_id: string
  skill_id?: string
  metric_name?: string
  severity?: string
}) {
  setClaimDrawerOpen(true)
  setClaimDrawerLoading(true)
  setClaimDrawerData(null)
  setClaimDrawerError(null)

  try {
    const res = await fetch(
      `/api/workspaces/${workspaceId}/reports/trace-claim`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          claim_id:     claim.claim_id,
          skill_run_id: claim.skill_run_id,
          document_id:  reportDocument?.id,
        })
      }
    )
    if (!res.ok) throw new Error('Failed to load evidence')
    const data = await res.json()
    setClaimDrawerData(data)
  } catch (err) {
    setClaimDrawerError('Could not load evidence for this claim.')
  } finally {
    setClaimDrawerLoading(false)
  }
}
```

### Step B — click handler

Add a click handler on the div that wraps the TipTap
editor output. Find the container div that renders the
TipTap content and add onClick:
```typescript
function handleEditorClick(event: React.MouseEvent) {
  const target = event.target as HTMLElement

  if (target.classList.contains('pandora-claim--hyperlink')) {
    const claimId    = target.dataset.claimId
    const skillRunId = target.dataset.skillRunId

    if (claimId && skillRunId) {
      event.preventDefault()
      event.stopPropagation()
      handleQuestionClaim({
        claim_id:     claimId,
        skill_run_id: skillRunId,
        skill_id:     target.dataset.skillId,
        metric_name:  target.dataset.metricName,
        severity:     target.dataset.severity,
      })
    }
  }
}
```

Attach to the editor container:
```typescript
<div onClick={handleEditorClick}>
  {/* existing TipTap editor JSX */}
</div>
```

One event listener on the container handles every
hyperlinked number in every section — no per-element
wiring needed.
