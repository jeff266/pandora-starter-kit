## Fix 1: Trim skill_runs output at write time

The skill runtime writes full skill output to the
output column. Transcripts and raw evidence are
the likely culprits.

In the skill runtime (wherever skill output is
persisted), add a trimming step before writing:
```typescript
function trimSkillOutput(output: any): any {
  if (!output || typeof output !== 'object')
    return output;

  const trimmed = { ...output };

  // Remove raw transcript text — it's in the
  // conversations table, no need to duplicate
  if (trimmed.transcript_text) {
    trimmed.transcript_text = '[trimmed]';
  }
  if (trimmed.trimmed_conversations) {
    delete trimmed.trimmed_conversations;
  }
  if (trimmed.recent_conversations) {
    // Keep count and metadata, not full text
    trimmed.recent_conversations =
      Array.isArray(trimmed.recent_conversations)
        ? trimmed.recent_conversations
            .slice(0, 3)
            .map((c: any) => ({
              id: c.id,
              title: c.title,
              started_at: c.started_at,
              duration_seconds: c.duration_seconds,
              // Remove transcript/summary text
            }))
        : trimmed.recent_conversations;
  }

  // Cap any array fields at 50 items
  for (const [key, val] of Object.entries(trimmed)) {
    if (Array.isArray(val) && val.length > 50) {
      (trimmed as any)[key] = val.slice(0, 50);
      console.log(
        `[SkillOutput] Trimmed ${key} from ` +
        `${val.length} to 50 items`
      );
    }
  }

  return trimmed;
}
```

Apply this function before every INSERT/UPDATE
to the output column in skill_runs.

Also add a DB-level size guard as a second line
of defense — log a warning if the serialized
output exceeds 2MB:
```typescript
const serialized = JSON.stringify(trimmedOutput);
if (serialized.length > 2_000_000) {
  console.warn(
    `[SkillRuntime] Output size warning: ` +
    `${Math.round(serialized.length / 1000)}KB ` +
    `for skill ${skillId}. Consider further trimming.`
  );
}
```

Do NOT backfill existing rows — the 231MB is
already written. Just prevent future growth.

## Fix 2: Normalize activities properties JSONB

The activities table stores raw HubSpot API
responses in properties JSONB. Each row is
~500KB because HubSpot responses include
nested objects Pandora never uses.

Find where activities are written during HubSpot
sync. Add a normalization step:
```typescript
function normalizeActivityProperties(
  raw: Record<string, any>
): Record<string, any> {
  // Keep only fields Pandora's skills actually use
  // Check tool-definitions.ts for queryActivities
  // to find which fields are read
  const KEEP_FIELDS = [
    'hs_activity_type',
    'hs_timestamp',
    'hs_body_preview',
    'hs_call_duration',
    'hs_call_status',
    'hs_call_direction',
    'hs_meeting_title',
    'hs_meeting_outcome',
    'hubspot_owner_id',
    'hs_email_subject',
    'hs_email_status',
  ];

  const normalized: Record<string, any> = {};
  for (const field of KEEP_FIELDS) {
    if (raw[field] !== undefined) {
      normalized[field] = raw[field];
    }
  }

  // Keep any non-nested scalar values not in list
  for (const [key, val] of Object.entries(raw)) {
    if (
      !KEEP_FIELDS.includes(key) &&
      typeof val !== 'object' &&
      val !== null
    ) {
      normalized[key] = val;
    }
  }

  return normalized;
}
```

Apply during HubSpot activities sync, not
retroactively. Existing bloated rows stay as-is —
only new syncs write normalized data.

## Migration: Add missing indexes

Create /migrations/XXX_performance_indexes.sql:
```sql
-- Close date range queries for BI
-- Used by: "deals closing this quarter",
--   "pipeline by close date bucket"
CREATE INDEX CONCURRENTLY IF NOT EXISTS
  idx_deals_close_date
  ON deals(workspace_id, close_date)
  WHERE close_date IS NOT NULL;

-- Pipeline + stage composite for Fellowship exclusion
-- Used by: getForecastPipelines() scoping
CREATE INDEX CONCURRENTLY IF NOT EXISTS
  idx_deals_pipeline_stage
  ON deals(workspace_id, pipeline, stage_normalized)
  WHERE amount > 0;

-- Amount range for deal value queries
-- Used by: "show me deals over $100K",
--   "pipeline by deal size bucket"
CREATE INDEX CONCURRENTLY IF NOT EXISTS
  idx_deals_amount
  ON deals(workspace_id, amount DESC)
  WHERE amount > 0;

-- Last activity date for staleness queries
-- Used by: "deals dark 30+ days",
--   chart intelligence stale deal extraction
CREATE INDEX CONCURRENTLY IF NOT EXISTS
  idx_deals_last_activity
  ON deals(workspace_id, last_activity_date)
  WHERE last_activity_date IS NOT NULL;

-- skill_runs: faster lookup for summarizer
-- The existing workspace_skill_created index is good
-- but add status filter for completed-only queries
CREATE INDEX CONCURRENTLY IF NOT EXISTS
  idx_skill_runs_workspace_skill_status
  ON skill_runs(workspace_id, skill_id, status,
    created_at DESC)
  WHERE status = 'completed';

-- report_charts: workspace + document lookup
-- Already have idx_report_charts_document
-- Add workspace + section for the data picker
CREATE INDEX CONCURRENTLY IF NOT EXISTS
  idx_report_charts_workspace_section
  ON report_charts(workspace_id, section_id,
    created_at DESC);
```

Use CONCURRENTLY so indexes build without
locking the table. Safe to run on live DB.
