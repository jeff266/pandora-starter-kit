function computeCoverageRatio(
  openPipeline: number,
  target: number,
  closedWon: number
): {
  ratio: number | null
  formula: string
  interpretation: string
} {
  const remainingGap = target - closedWon

  // Quota already exceeded
  if (closedWon >= target) {
    return {
      ratio: null,
      formula: 'Quota exceeded — coverage not applicable',
      interpretation:
        `Closed $${(closedWon/1000).toFixed(0)}K against ` +
        `$${(target/1000).toFixed(0)}K target. ` +
        `Coverage is a forward-looking metric — ` +
        `focus shifts to upside and next quarter.`
    }
  }

  // Normal case: coverage of remaining gap
  const ratio = openPipeline / remainingGap
  return {
    ratio,
    formula:
      `Open pipeline ($${(openPipeline/1000).toFixed(0)}K) / ` +
      `Remaining gap ($${(remainingGap/1000).toFixed(0)}K) ` +
      `= ${ratio.toFixed(2)}x`,
    interpretation:
      ratio >= 3.0
        ? `Strong coverage — ${ratio.toFixed(1)}x against ` +
          `remaining $${(remainingGap/1000).toFixed(0)}K gap.`
        : ratio >= 2.0
        ? `Adequate coverage — ${ratio.toFixed(1)}x but ` +
          `monitor for slippage.`
        : `Thin coverage — ${ratio.toFixed(1)}x against ` +
          `remaining gap. At risk.`
  }
}
