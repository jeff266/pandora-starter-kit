## Task 2: report_annotations Migration

Create the next migration after 186:

```sql
CREATE TABLE IF NOT EXISTS report_annotations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  workspace_id UUID NOT NULL REFERENCES workspaces(id)
    ON DELETE CASCADE,
  report_document_id UUID NOT NULL
    REFERENCES report_documents(id) ON DELETE CASCADE,
  section_id TEXT NOT NULL,
  paragraph_index INTEGER NOT NULL DEFAULT 0,
  annotation_type TEXT NOT NULL
    CHECK (annotation_type IN ('note', 'override', 'flag')),
  content TEXT NOT NULL DEFAULT '',
  original_content TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_report_annotations_report
  ON report_annotations(report_document_id);

CREATE INDEX idx_report_annotations_workspace
  ON report_annotations(workspace_id);
```

## Task 5: Tab Navigation Between Paragraphs

In AnnotatableSection, handle Tab key on paragraph elements:

```typescript
const handleKeyDown = (
  e: React.KeyboardEvent<HTMLParagraphElement>,
  index: number
) => {
  if (!isAnnotating) return;

  if (e.key === 'Tab' && !e.shiftKey) {
    e.preventDefault();
    const allParagraphs = document.querySelectorAll(
      '[data-paragraph]'
    );
    const currentIndex = Array.from(allParagraphs)
      .findIndex(el =>
        el.getAttribute('data-section') === section.id &&
        el.getAttribute('data-paragraph') === String(index)
      );
    const next = allParagraphs[currentIndex + 1] as HTMLElement;
    if (next) next.focus();
  }
};
```

This navigates across section boundaries automatically
because all paragraphs across all sections share the same
[data-paragraph] selector.

```typescript
const [annotations, setAnnotations] = useState<Annotation[]>([]);

      .then(r => r.json())
      .then(setAnnotations);

{reportDocument.sections.map(section => (
  <AnnotatableSection
    key={section.id}
    section={section}
    annotations={annotations.filter(
      a => a.section_id === section.id
    )}
    isAnnotating={isAnnotating}
    onAnnotationSave={async (annotation) => {
      const res = await fetch(
        `/api/workspaces/${workspaceId}/reports/
          ${reportDocument.id}/annotations`,
        {
          method: 'POST',
          headers: {'Content-Type': 'application/json'},
          body: JSON.stringify(annotation),
        }
      );
      const saved = await res.json();
      setAnnotations(prev => [
        ...prev.filter(a =>
          !(a.section_id === annotation.section_id &&
            a.paragraph_index === annotation.paragraph_index)
        ),
        saved,
      ]);
    }}
    onAnnotationDelete={async (annotationId) => {
      await fetch(
        `/api/workspaces/${workspaceId}/reports/
          ${reportDocument.id}/annotations/${annotationId}`,
        { method: 'DELETE' }
      );
      setAnnotations(prev =>
        prev.filter(a => a.id !== annotationId)
      );
    }}
  />
))}
```
