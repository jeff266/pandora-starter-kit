## Build 3: AddToReportButton component

Create client/src/components/chat/
AddToReportButton.tsx

```typescript
interface AddToReportButtonProps {
  chart: {
    spec: any;
    png_base64: string;
    suggested_section_id?: string;
  };
  workspaceId: string;
  token: string;
}

type ButtonState =
  'idle' | 'picking' | 'saving' | 'saved' | 'error';
```

State:
```typescript
const [state, setState] =
  useState<ButtonState>('idle');
const [reportId, setReportId] =
  useState<string | null>(null);
const [sections, setSections] =
  useState<{ id: string; title: string }[]>([]);
const [selectedSection, setSelectedSection] =
  useState(chart.suggested_section_id || '');
```

On click (idle → picking):
```typescript
async function handleClick() {
  setState('picking');
  try {
    const res = await fetch(
      `/api/workspaces/${workspaceId}/reports/current`,
      { headers: { Authorization: `Bearer ${token}` } }
    );
    if (!res.ok) {
      // No report exists — hide button
      setState('idle');
      return;
    }
    const report = await res.json();
    setReportId(report.id);

    // Parse sections from report
    // sections may be stored as array of
    // { id, title } objects
    const sectionList = Array.isArray(report.sections)
      ? report.sections.map((s: any) => ({
          id: s.id || s.section_id,
          title: s.title,
        }))
      : [
          { id: 'node-deal-execution',
            title: 'Deal Execution' },
          { id: 'node-pipeline-conv',
            title: 'Pipeline Conversion' },
          { id: 'node-team-execution',
            title: 'Team Execution' },
          { id: 'node-forward-look',
            title: 'Forward Look' },
        ];

    setSections(sectionList);

    // Pre-select suggested or first section
    if (!selectedSection && sectionList.length > 0) {
      setSelectedSection(
        chart.suggested_section_id
        || sectionList[0].id
      );
    }
  } catch {
    setState('error');
  }
}
```

On save:
```typescript
async function handleSave() {
  if (!reportId || !selectedSection) return;
  setState('saving');

  try {
    const res = await fetch(
      `/api/workspaces/${workspaceId}/reports/` +
      `${reportId}/charts`,
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          Authorization: `Bearer ${token}`,
        },
        body: JSON.stringify({
          section_id: selectedSection,
          chart_type: chart.spec?.chart_type
            || 'bar',
          title: chart.spec?.title || 'Chart',
          data: (chart.spec?.data_points || [])
            .map((d: any) => ({
              label: d.label,
              value: d.value,
            })),
        }),
      }
    );

    if (!res.ok) throw new Error('Save failed');

    setState('saved');
    // Revert to idle after 2 seconds
    setTimeout(() => setState('idle'), 2000);

  } catch {
    setState('error');
  }
}
```

   from Ask Pandora is embedded in the section

   opens with chart title as context

9. If no report exists → "Add to report"
   button hidden entirely

    text responses and existing report charts
