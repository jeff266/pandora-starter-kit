## Data loading

Create server/routes/chart-data.ts:
```typescript
// GET /:workspaceId/chart-data/sources
// Returns available data sources with record counts

router.get(
  '/:workspaceId/chart-data/sources',
  requireAuth,
  async (req, res) => {
    const { workspaceId } = req.params;
    const CHARTABLE_SKILLS = [
      'deal-risk-review',
      'pipeline-hygiene',
      'pipeline-coverage',
      'forecast-rollup',
    ];

    const results = await query(`
      SELECT DISTINCT ON (skill_id)
        skill_id,
        id as run_id,
        created_at,
        jsonb_array_length(
          output->'evidence'->'evaluated_records'
        ) as record_count
      FROM skill_runs
      WHERE workspace_id = $1
        AND status = 'completed'
        AND skill_id = ANY($2)
        AND output->'evidence'->'evaluated_records'
          IS NOT NULL
        AND created_at > NOW() - INTERVAL '7 days'
      ORDER BY skill_id, created_at DESC
    `, [workspaceId, CHARTABLE_SKILLS]);

    res.json({ sources: results.rows });
  }
);

// GET /:workspaceId/chart-data/:skillId
// Returns evaluated_records for a skill

router.get(
  '/:workspaceId/chart-data/:skillId',
  requireAuth,
  async (req, res) => {
    const { workspaceId, skillId } = req.params;

    const result = await query(`
      SELECT
        output->'evidence'->'evaluated_records'
          as records,
        output->'evidence'->'claims'
          as claims,
        created_at
      FROM skill_runs
      WHERE workspace_id = $1
        AND skill_id = $2
        AND status = 'completed'
        AND created_at > NOW() - INTERVAL '7 days'
      ORDER BY created_at DESC
      LIMIT 1
    `, [workspaceId, skillId]);

    if (!result.rows[0]) {
      return res.status(404).json(
        { error: 'No recent skill run found' }
      );
    }

    res.json({
      records: result.rows[0].records || [],
      claims: result.rows[0].claims || [],
      fetched_at: result.rows[0].created_at,
    });
  }
);
```

// Safe field extractor — handles nested paths
// and type coercion
export function extractField(
  record: any,
  fieldKey: string,
  fieldDef: DataSourceField
): string | number {
  const val = fieldKey.split('.').reduce(
    (obj, key) => obj?.[key], record
  );

  if (val === null || val === undefined) return 0;

  if (fieldDef.parse === 'float') {
    return parseFloat(String(val)) || 0;
  }
  if (fieldDef.parse === 'int') {
    return parseInt(String(val), 10) || 0;
  }
  if (fieldDef.type === 'currency' ||
      fieldDef.type === 'number') {
    return Number(val) || 0;
  }
  return String(val);
}

## SEMANTIC COLOR RESOLUTION
```typescript
// client/src/lib/chartColors.ts

export const SEMANTIC_COLORS = {
  // Severity
  critical: '#EF4444',
  warning:  '#F59E0B',
  healthy:  '#0D9488',
  high:     '#EF4444',
  medium:   '#F59E0B',
  low:      '#0D9488',

  // Stale (days since activity thresholds)
  stale:    '#EF4444',
  fresh:    '#0D9488',

  // Forecast categories
  closed:     '#0D9488',
  commit:     '#0D9488',
  best_case:  '#F59E0B',
  pipeline:   '#CBD5E1',
  omitted:    '#94A3B8',

  // Defaults
  uniform:  '#0D9488',
  fallback: '#CBD5E1',
};

export function resolveColor(
  record: any,
  colorScheme: string,
  dataSource: DataSourceDef,
  uniformColor = '#0D9488'
): string {

  if (colorScheme === 'uniform') return uniformColor;

  if (colorScheme === 'semantic') {
    // Try severity first
    const severity = record.severity
      || record.flags?.risk_level
      || record.flags?.stale_flag;

    if (severity && SEMANTIC_COLORS[severity as keyof
        typeof SEMANTIC_COLORS]) {
      return SEMANTIC_COLORS[severity as keyof
        typeof SEMANTIC_COLORS];
    }

    // Stale threshold: >30 days = red, >14 = amber
    const days = record.fields?.days_since_activity;
    if (typeof days === 'number') {
      if (days > 100) return SEMANTIC_COLORS.critical;
      if (days > 30) return SEMANTIC_COLORS.warning;
      return SEMANTIC_COLORS.healthy;
    }
  }

  if (colorScheme === 'categorical') {
    const cat = record.fields?.forecast_category;
    if (cat && SEMANTIC_COLORS[cat as keyof
        typeof SEMANTIC_COLORS]) {
      return SEMANTIC_COLORS[cat as keyof
        typeof SEMANTIC_COLORS];
    }
  }

  return uniformColor;
}
```

// client/src/components/report/ContextMenu.tsx

interface ContextMenuProps {
  x: number;
  y: number;
  onNote: () => void;
  onChart: () => void;
  onFlag: () => void;
  onClose: () => void;
}

export function ContextMenu({
  x, y, onNote, onChart, onFlag, onClose
}: ContextMenuProps) {
  useEffect(() => {
    const handler = () => onClose();
    window.addEventListener('click', handler);
    return () =>
      window.removeEventListener('click', handler);
  }, [onClose]);

  return (
    <div
      style={{
        position: 'fixed',
        left: x,
        top: y,
        zIndex: 1000,
        background: 'white',
        border: '0.5px solid #E2E8F0',
        borderRadius: 8,
        boxShadow: '0 4px 20px rgba(0,0,0,0.12)',
        padding: '4px 0',
        minWidth: 160,
        fontFamily: '-apple-system, sans-serif',
      }}
      onClick={e => e.stopPropagation()}
    >
      {[
        { icon: '✎', label: 'Add note',
          action: onNote },
        { icon: '▤', label: 'Insert chart',
          action: onChart },
        { icon: '⚑', label: 'Flag for review',
          action: onFlag },
      ].map(item => (
        <button
          key={item.label}
          onClick={item.action}
          style={{
            display: 'flex',
            alignItems: 'center',
            gap: 10,
            width: '100%',
            padding: '8px 14px',
            background: 'none',
            border: 'none',
            fontSize: 13,
            color: '#374151',
            cursor: 'pointer',
            textAlign: 'left',
          }}
          onMouseEnter={e => {
            (e.target as HTMLElement)
              .style.background = '#F8FAFC';
          }}
          onMouseLeave={e => {
            (e.target as HTMLElement)
              .style.background = 'none';
          }}
        >
          <span style={{ fontSize: 14,
            color: '#64748B', width: 16 }}>
            {item.icon}
          </span>
          {item.label}
        </button>
      ))}
    </div>
  );
}

## VALIDATIONS
```typescript
interface Validation {
  type: 'error' | 'warning';
  message: string;
  action?: { label: string; fn: () => void };
}

function runValidations(
  title: string,
  chartType: string,
  records: any[],
  field: DataSourceField,
  colorScheme: string,
): Validation[] {

  const v: Validation[] = [];

  // Title ends with question mark
  if (title.trim().endsWith('?')) {
    v.push({
      type: 'warning',
      message: 'Chart titles should state a ' +
        'conclusion, not ask a question.',
      action: {
        label: 'Remove ?',
        fn: () => setChartTitle(
          title.trim().slice(0, -1)
        ),
      },
    });
  }

  // Line chart with < 3 points
  if (chartType === 'line' && records.length < 3) {
    v.push({
      type: 'error',
      message: 'Line charts need 3+ data points ' +
        'to show a meaningful trend.',
      action: {
        label: 'Switch to bar',
        fn: () => setChartType('bar'),
      },
    });
  }

  // Donut with too many segments
  if (chartType === 'donut' && records.length > 5) {
    v.push({
      type: 'warning',
      message: `Donut with ${records.length} ` +
        'segments is hard to read. ' +
        'Showing top 5 is cleaner.',
      action: {
        label: 'Limit to 5',
        fn: () => setLimit(5),
      },
    });
  }

  // Currency field with % format
  if (field.type === 'currency' &&
      numberFormat === 'pct') {
    v.push({
      type: 'error',
      message: 'Percentage format doesn\'t make ' +
        'sense for currency values.',
      action: {
        label: 'Switch to $K',
        fn: () => setNumberFormat('K'),
      },
    });
  }

  // Coverage ratio = 0 warning
  if (field.key.includes('coverage_ratio') ||
      field.key.includes('gap_to_quota')) {
    v.push({
      type: 'warning',
      message: 'Revenue targets not configured — ' +
        'this chart will show 0 values.',
    });
  }

  // Empty title
  if (!title.trim()) {
    v.push({
      type: 'error',
      message: 'Chart title is required.',
    });
  }

  return v;
}
```

## HELPER FUNCTIONS
```typescript
function formatValue(
  val: number,
  format: 'K'|'M'|'raw'|'pct',
  type: string
): string {
  if (type === 'string' || type === 'category')
    return String(val);

  if (format === 'K') {
    return val >= 1000
      ? `$${(val/1000).toFixed(0)}K`
      : `$${val}`;
  }
  if (format === 'M') {
    return val >= 1000000
      ? `$${(val/1000000).toFixed(1)}M`
      : val >= 1000
      ? `$${(val/1000).toFixed(0)}K`
      : `$${val}`;
  }
  if (format === 'pct') {
    return `${val.toFixed(1)}%`;
  }
  return val.toLocaleString();
}

function buildChartSpec(
  records: any[],
  field: DataSourceField,
  source: DataSourceDef,
  config: {
    chartType: string;
    colorScheme: string;
    showLegend: boolean;
    numberFormat: string;
    title: string;
  }
): ChartNodeSpec {
  // Sort records
  const sorted = [...records].sort((a, b) => {
    const aVal = extractField(a, sortField, field);
    const bVal = extractField(b, sortField, field);
    return sortDir === 'desc'
      ? Number(bVal) - Number(aVal)
      : Number(aVal) - Number(bVal);
  }).slice(0, limit);

  const dataPoints = sorted.map(r => ({
    label: String(
      r.entity_name || r[source.nameField] || ''
    ).split(' - ')[0].trim(), // strip product suffix
    value: Math.round(
      Number(extractField(r, field.key, field)) /
      (config.numberFormat === 'K' ? 1000 : 1)
    ),
    semantic_tag: getSemanticTag(r, source),
  }));

  return {
    should_chart: true,
    chart_type: config.chartType as any,
    title: config.title,
    data_points: dataPoints,
    color_scheme: config.colorScheme as any,
    show_legend: config.showLegend,
  };
}

function getSemanticTag(
  record: any,
  source: DataSourceDef
): string {
  if (source.id === 'at_risk_deals') {
    const score = Number(
      record.fields?.risk_score || 0
    );
    if (score > 80) return 'dead';
    if (score > 60) return 'at_risk';
    return 'healthy';
  }
  if (source.id === 'stale_deals') {
    const days = record.fields?.days_since_activity
      || 0;
    if (days > 100) return 'dead';
    if (days > 30) return 'at_risk';
    return 'healthy';
  }
  if (source.id === 'forecast_pipeline') {
    const cat = record.fields?.forecast_category;
    if (cat === 'closed' || cat === 'commit')
      return 'positive';
    if (cat === 'best_case') return 'at_risk';
    return 'neutral';
  }
  return 'neutral';
}
```
