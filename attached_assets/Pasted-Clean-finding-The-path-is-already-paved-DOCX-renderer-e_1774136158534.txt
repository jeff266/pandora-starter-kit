## Part 1 — Server Endpoint

File: server/routes/reports.ts

Add one new route, mirroring the existing export endpoint pattern:

```typescript
// POST /:workspaceId/reports/:reportId/documents/:documentId/export/google-docs
router.post(
  '/:workspaceId/reports/:reportId/documents/:documentId/export/google-docs',
  requireAuth,
  async (req, res) => {
    const { workspaceId, reportId, documentId } = req.params

    try {
      // 1. Check Google Drive credentials for this workspace
      const credentials = await getWorkspaceGoogleCredentials(workspaceId)
      if (!credentials) {
        return res.status(400).json({
          error: 'google_drive_not_connected',
          message: 'Connect Google Drive in Settings → Integrations first.',
          settings_url: '/settings/integrations',
        })
      }

      // 2. Load the report document
      const doc = await db.query(
        `SELECT * FROM report_documents
         WHERE id = $1 AND workspace_id = $2`,
        [documentId, workspaceId]
      )
      if (!doc.rows.length) {
        return res.status(404).json({ error: 'Document not found' })
      }
      const reportDocument = doc.rows[0]

      // 3. Render as DOCX buffer
      // Use the same renderDOCX() call the existing export endpoint uses.
      // Pass reportDocument — follow the exact same context shape.
      const docxBuffer = await renderDOCX(reportDocument, workspaceId)

      // 4. Compose a clean filename
      const title = [
        reportDocument.document_type?.toUpperCase() ?? 'Report',
        reportDocument.week_label ?? '',
      ].filter(Boolean).join(' — ')
      // e.g. "WBR — Week of March 17, 2026"

      // 5. Upload to Google Drive with DOCX→Google Doc conversion
      const freshCredentials = await ensureFreshToken(credentials)

      const uploaded = await GoogleDriveClient.uploadFile(
        freshCredentials,
        null,          // no specific folder — root Drive
        `${title}.docx`,
        docxBuffer,
        'application/vnd.google-apps.document'
        // This mimeType triggers server-side DOCX → Google Doc conversion
      )

      if (!uploaded?.id) {
        return res.status(500).json({
          error: 'Upload failed',
          message: 'Google Drive upload did not return a file ID.',
        })
      }

      // 6. Return the Google Doc URL
      const docUrl = `https://docs.google.com/document/d/${uploaded.id}/edit`

      return res.json({
        doc_url: docUrl,
        doc_id: uploaded.id,
        title,
      })

    } catch (err) {
      console.error('[google-docs-export] failed:', err)
      return res.status(500).json({
        error: 'Export failed',
        message: (err as Error).message,
      })
    }
  }
)
```

### 2c — Add state and handler

```typescript
const [exportingToGoogleDocs, setExportingToGoogleDocs] =
  useState(false)
const [googleDocsUrl, setGoogleDocsUrl] = useState<string | null>(null)

async function handleExportToGoogleDocs() {
  if (!reportDocument?.id) return
  setExportingToGoogleDocs(true)

  try {
    const res = await fetch(
      `/api/workspaces/${workspaceId}/reports/${reportId}` +
      `/documents/${reportDocument.id}/export/google-docs`,
      { method: 'POST' }
    )
    const data = await res.json()

    if (!res.ok) {
      if (data.error === 'google_drive_not_connected') {
        // Show a connect prompt
        setGoogleDocsError(
          'Google Drive not connected. ' +
          'Go to Settings → Integrations to connect.'
        )
      } else {
        setGoogleDocsError(data.message ?? 'Export failed')
      }
      return
    }

    setGoogleDocsUrl(data.doc_url)
    // Open in new tab immediately
    window.open(data.doc_url, '_blank', 'noopener,noreferrer')

  } catch (err) {
    setGoogleDocsError('Export failed. Please try again.')
  } finally {
    setExportingToGoogleDocs(false)
  }
}

const [googleDocsError, setGoogleDocsError] = useState<string | null>(null)
```
