### ClaimMark.ts update

In renderHTML(), add numeric detection to apply the

```typescript
renderHTML({ HTMLAttributes }) {
  const NUMERIC_PATTERNS = [
    /\$[\d,.]+[MKB]?/,
    /\d[\d,]*\s+deals?/,
    /\d[\d,]*\s+accounts?/,
    /[\d.]+%/,
    /[\d.]+x/,
    /\d[\d,]*\s+reps?/,
    /\d+\s+days?/,
    /\d[\d,]*/,  // any standalone number
  ]

  // We can't access the text content in renderHTML,
  // so add a data-numeric attribute in the post-processor
  // instead (see note below)
  const isNumeric = HTMLAttributes['data-numeric'] === 'true'

  return [
    'span',
    mergeAttributes(HTMLAttributes, {
      'data-claim-id':     HTMLAttributes.claim_id,
      'data-skill-id':     HTMLAttributes.skill_id,
      'data-skill-run-id': HTMLAttributes.skill_run_id,
      'data-metric-name':  HTMLAttributes.metric_name,
      'data-severity':     HTMLAttributes.severity,
      'class': isNumeric
        ? 'pandora-claim pandora-claim--hyperlink'
        : 'pandora-claim',
    }),
    0
  ]
},
```

```typescript
function handleEditorClick(event: React.MouseEvent) {
  const target = event.target as HTMLElement
  if (target.classList.contains('pandora-claim--hyperlink')) {
    const claimId    = target.dataset.claimId
    const skillRunId = target.dataset.skillRunId
    if (claimId && skillRunId) {
      event.preventDefault()
      event.stopPropagation()
      handleQuestionClaim({
        claim_id:     claimId,
        skill_id:     target.dataset.skillId ?? '',
        skill_run_id: skillRunId,
        metric_name:  target.dataset.metricName ?? '',
        severity:     target.dataset.severity ?? 'info',
      })
    }
  }
}
```

## Part 4 — handleQuestionClaim function

Shared by both the click handler (Build 7) and the
selection handler (Build 5):

```typescript
async function handleQuestionClaim(claim: ClaimAttrs) {
  setClaimDrawerOpen(true)
  setClaimDrawerLoading(true)
  setClaimDrawerData(null)
  setClaimDrawerError(null)

  try {
    const res = await fetch(
      `/api/workspaces/${workspaceId}/reports/trace-claim`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          claim_id:     claim.claim_id,
          skill_run_id: claim.skill_run_id,
          document_id:  reportDocument.id,
        })
      }
    )
    if (!res.ok) throw new Error('Failed to load evidence')
    const data = await res.json()
    setClaimDrawerData(data)
  } catch (err) {
    setClaimDrawerError('Could not load evidence for this claim.')
  } finally {
    setClaimDrawerLoading(false)
  }
}
```

## Part 6 — Wire Resolution Paths

In ReportViewer.tsx, define the two resolution handlers
and pass them to ClaimEvidenceDrawer:

```typescript
function handleChallengeClaim(data: TraceClaimResult) {
  const prompt =
    `I want to challenge this claim: "${data.claim_text}". ` +
    `The ${formatSkillName(data.skill_id)} skill reported ` +
    `${data.metric_values?.[0]} ${data.metric_name}. ` +
    `Challenge it from both sides — what supports it and ` +
    `what would make it wrong?`

  openPandoraRail({
    prompt,
    forcedMode: 'prosecutor_defense',
  })
  setClaimDrawerOpen(false)
}

function handleRecalibrateClaim(data: TraceClaimResult) {
  const prompt =
    `The definition behind this number might be wrong: ` +
    `"${data.claim_text}". ` +
    `Current filter: ${data.dimension_summary ?? 'default definition'}. ` +
    `Let's adjust the ${data.metric_name} definition.`

  openPandoraRail({
    prompt,
    intent: 'calibration_interview',
  })
  setClaimDrawerOpen(false)
}
```
