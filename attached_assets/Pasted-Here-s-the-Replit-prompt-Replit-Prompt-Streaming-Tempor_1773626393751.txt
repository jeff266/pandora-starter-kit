### What to build

A streaming text greeting that appears at the top of the Concierge page when it loads. It types itself out character by character, then the rest of the page content fades in beneath it. Two to three sentences maximum. Never the same twice.

### Part 1 — Greeting generator endpoint

```
GET /api/workspaces/:id/concierge/greeting
```

Returns a greeting string. Called on every Concierge page load. Fast — reads from cached skill run data, no fresh LLM call needed for most inputs. The greeting is assembled from structured data, not generated by Claude on every load.

```typescript
// server/routes/concierge-greeting.ts

export async function generateGreeting(
  workspaceId: string,
  userId: string,
  userRole: 'admin' | 'rep'
): Promise<string> {

  // ── TEMPORAL CONTEXT ──
  const now = new Date()
  const hour = now.getHours()
  const dayOfWeek = now.getDay()  // 0 = Sunday, 1 = Monday

  const timeGreeting =
    hour < 12 ? 'Good morning' :
    hour < 17 ? 'Good afternoon' :
    'Good evening'

  const dayContext =
    dayOfWeek === 1 ? 'Monday' :
    dayOfWeek === 5 ? 'Friday' :
    null

  // ── QUARTER CONTEXT ──
  const quarterStart = getQuarterStart(now)
  const quarterEnd = getQuarterEnd(now)
  const totalWeeks = Math.ceil(
    (quarterEnd.getTime() - quarterStart.getTime()) / (7 * 24 * 60 * 60 * 1000)
  )
  const currentWeek = Math.ceil(
    (now.getTime() - quarterStart.getTime()) / (7 * 24 * 60 * 60 * 1000)
  )
  const daysRemaining = Math.ceil(
    (quarterEnd.getTime() - now.getTime()) / (1000 * 60 * 60 * 24)
  )
  const quarterLabel = getQuarterLabel(now)  // "Q1 2026"

  const quarterUrgency =
    currentWeek <= 3 ? 'early' :
    currentWeek <= 8 ? 'mid' :
    daysRemaining <= 14 ? 'closing' :
    'late'

  // ── PERFORMANCE CONTEXT ──
  // Read from latest forecast-rollup skill run — no recompute
  const forecastRun = await getLatestSkillRun(workspaceId, 'forecast-rollup')
  const attainmentPct = forecastRun?.result_data?.metrics?.attainmentPct ?? null
  const closedWon = forecastRun?.result_data?.metrics?.closedWonValue ?? null
  const quota = forecastRun?.result_data?.metrics?.teamQuota ?? null

  // ── SPRINT CONTEXT ──
  const sprintActions = await query(`
    SELECT COUNT(*) as total,
      COUNT(*) FILTER (WHERE severity = 'critical') as critical
    FROM actions
    WHERE workspace_id = $1
      AND sprint_week = date_trunc('week', NOW())
      AND state = 'pending'
  `, [workspaceId])

  const pendingActions = parseInt(sprintActions.rows[0]?.total ?? '0')
  const criticalActions = parseInt(sprintActions.rows[0]?.critical ?? '0')

  // ── HYPOTHESIS CONTEXT ──
  const trippedHypotheses = await query(`
    SELECT COUNT(*) as count
    FROM standing_hypotheses
    WHERE workspace_id = $1
      AND status = 'active'
      AND (
        (alert_direction = 'below' AND current_value < alert_threshold)
        OR (alert_direction = 'above' AND current_value > alert_threshold)
      )
  `, [workspaceId])
  const alertCount = parseInt(trippedHypotheses.rows[0]?.count ?? '0')

  // ── ASSEMBLE GREETING ──
  // Role-aware: admin sees team view, rep sees personal view

  if (userRole === 'rep') {
    return assembleRepGreeting({
      timeGreeting, dayContext, quarterLabel, currentWeek,
      totalWeeks, daysRemaining, quarterUrgency,
      attainmentPct, closedWon, quota, pendingActions
    })
  }

  return assembleAdminGreeting({
    timeGreeting, dayContext, quarterLabel, currentWeek,
    totalWeeks, daysRemaining, quarterUrgency,
    attainmentPct, closedWon, quota,
    pendingActions, criticalActions, alertCount
  })
}

function assembleAdminGreeting(ctx: AdminGreetingContext): string {
  const lines: string[] = []

  // Line 1: temporal + quarter state
  if (ctx.quarterUrgency === 'closing') {
    lines.push(
      `${ctx.timeGreeting}. ${ctx.quarterLabel} closes in ${ctx.daysRemaining} days.`
    )
  } else if (ctx.dayContext === 'Monday') {
    lines.push(
      `${ctx.timeGreeting}. Week ${ctx.currentWeek} of ${ctx.totalWeeks} in ${ctx.quarterLabel}.`
    )
  } else {
    lines.push(
      `${ctx.timeGreeting}. ${ctx.quarterLabel}, week ${ctx.currentWeek}.`
    )
  }

  // Line 2: performance state (only if quota data available)
  if (ctx.attainmentPct !== null && ctx.closedWon !== null) {
    if (ctx.attainmentPct >= 100) {
      lines.push(
        `${ctx.quarterLabel} is won at ${Math.round(ctx.attainmentPct)}% — ` +
        `${formatCurrency(ctx.closedWon)} closed against target.`
      )
    } else if (ctx.attainmentPct >= 75) {
      const gap = ctx.quota - ctx.closedWon
      lines.push(
        `Attainment is at ${Math.round(ctx.attainmentPct)}%. ` +
        `${formatCurrency(gap)} remaining to close.`
      )
    } else if (ctx.quarterUrgency === 'closing') {
      const gap = ctx.quota - ctx.closedWon
      lines.push(
        `${formatCurrency(gap)} to close in ${ctx.daysRemaining} days. ` +
        `The sprint has ${ctx.criticalActions} critical actions.`
      )
    }
  }

  // Line 3: sprint or alert state
  if (ctx.alertCount > 0 && ctx.quarterUrgency !== 'early') {
    lines.push(
      `${ctx.alertCount} hypothesis ${ctx.alertCount === 1 ? 'threshold has' : 'thresholds have'} ` +
      `tripped this week.`
    )
  } else if (ctx.pendingActions > 0 && ctx.dayContext === 'Monday') {
    lines.push(
      `${ctx.pendingActions} sprint ${ctx.pendingActions === 1 ? 'action' : 'actions'} queued for the week.`
    )
  }

  return lines.join(' ')
}

function assembleRepGreeting(ctx: RepGreetingContext): string {
  // Simpler — rep sees their own attainment and their own pending actions
  const lines: string[] = []

  lines.push(
    ctx.quarterUrgency === 'closing'
      ? `${ctx.timeGreeting}. ${ctx.daysRemaining} days left in ${ctx.quarterLabel}.`
      : `${ctx.timeGreeting}. Week ${ctx.currentWeek} of ${ctx.totalWeeks}.`
  )

  if (ctx.attainmentPct !== null) {
    lines.push(`You are at ${Math.round(ctx.attainmentPct)}% of target.`)
  }

  if (ctx.pendingActions > 0) {
    lines.push(
      `${ctx.pendingActions} ${ctx.pendingActions === 1 ? 'action' : 'actions'} in your sprint this week.`
    )
  }

  return lines.join(' ')
}

// Helper — format like "$481K" or "$1.2M"
function formatCurrency(value: number): string {
  if (value >= 1_000_000) return `$${(value / 1_000_000).toFixed(1)}M`
  if (value >= 1_000) return `$${Math.round(value / 1_000)}K`
  return `$${Math.round(value)}`
}
```

### Part 2 — Streaming component

In the Concierge page, replace the static header text with this component:

```tsx
// components/StreamingGreeting.tsx

import { useEffect, useState, useRef } from 'react'

interface StreamingGreetingProps {
  workspaceId: string
  onComplete?: () => void  // fires when streaming finishes — use to fade in page content
}

export function StreamingGreeting({ workspaceId, onComplete }: StreamingGreetingProps) {
  const [displayText, setDisplayText] = useState('')
  const [isComplete, setIsComplete] = useState(false)
  const [greeting, setGreeting] = useState<string | null>(null)
  const indexRef = useRef(0)

  // Fetch greeting on mount
  useEffect(() => {
    fetch(`/api/workspaces/${workspaceId}/concierge/greeting`)
      .then(r => r.json())
      .then(data => setGreeting(data.greeting))
      .catch(() => setGreeting(null))  // fail silently — page loads without greeting
  }, [workspaceId])

  // Start streaming once greeting is fetched
  useEffect(() => {
    if (!greeting) return

    indexRef.current = 0
    setDisplayText('')
    setIsComplete(false)

    // Variable speed — faster for spaces, punctuation pauses
    function getDelay(char: string): number {
      if (char === '.') return 280   // pause at sentence end
      if (char === ',') return 120   // brief pause at comma
      if (char === ' ') return 25    // fast through spaces
      return 22 + Math.random() * 18 // 22–40ms per character — natural variation
    }

    function typeNextChar() {
      if (indexRef.current >= greeting.length) {
        setIsComplete(true)
        onComplete?.()
        return
      }

      const char = greeting[indexRef.current]
      setDisplayText(prev => prev + char)
      indexRef.current++

      setTimeout(typeNextChar, getDelay(char))
    }

    // Brief pause before starting — page has just loaded
    const startTimer = setTimeout(typeNextChar, 300)
    return () => clearTimeout(startTimer)
  }, [greeting, onComplete])

  if (!greeting) return null

  return (
    <div style={{
      fontSize: '18px',
      fontWeight: 400,
      lineHeight: 1.5,
      color: 'var(--color-text-primary)',
      letterSpacing: '-0.01em',
      minHeight: '2.5rem',  // prevents layout shift
    }}>
      {displayText}
      {/* Blinking cursor while typing */}
      {!isComplete && (
        <span style={{
          display: 'inline-block',
          width: '2px',
          height: '1.1em',
          background: 'var(--color-text-tertiary)',
          marginLeft: '2px',
          verticalAlign: 'text-bottom',
          animation: 'blink 0.8s step-end infinite',
        }} />
      )}
      <style>{`
        @keyframes blink {
          0%, 100% { opacity: 1; }
          50% { opacity: 0; }
        }
      `}</style>
    </div>
  )
}
```

### Part 3 — Wire into Concierge page

In the Concierge page component, replace the static temporal header with `StreamingGreeting`. Use the `onComplete` callback to trigger a fade-in on the deal cards and briefing content below:

```tsx
const [contentVisible, setContentVisible] = useState(false)

// In the render:
<div style={{ marginBottom: '2rem' }}>
  <StreamingGreeting
    workspaceId={workspaceId}
    onComplete={() => setContentVisible(true)}
  />
</div>

<div style={{
  opacity: contentVisible ? 1 : 0,
  transform: contentVisible ? 'translateY(0)' : 'translateY(8px)',
  transition: 'opacity 0.4s ease, transform 0.4s ease',
}}>
  {/* existing Concierge content — briefing, deal cards, sprint */}
</div>
```

The deal cards and sprint content are invisible and slightly shifted down until the greeting finishes typing. Then they fade in and slide up. 400ms transition — subtle, not theatrical.
