```typescript
async function handleChangeType(newType: ChartType) {
  if (!suggestion && !acceptedChart) return;
  setSelectedType(newType);

  const source = acceptedChart || suggestion!;

  const res = await fetch(
    `/api/workspaces/${workspaceId}/reports/` +
    `${reportDocumentId}/charts`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${token}`,
      },
      body: JSON.stringify({
        section_id: sectionId,
        chart_type: newType,
        title: source.title,
        data: source.data,
        x_label: source.x_label,
        y_label: source.y_label,
      }),
    }
  );

  const saved = await res.json();
  setAcceptedChart(saved);

  // Revoke old blob URL
  if (previewUrl) URL.revokeObjectURL(previewUrl);

  // Load new PNG
  const imgRes = await fetch(
    `/api/workspaces/${workspaceId}/reports/` +
    `${reportDocumentId}/charts/${saved.id}/image`,
    { headers: { Authorization: `Bearer ${token}` } }
  );
  const blob = await imgRes.blob();
  setPreviewUrl(URL.createObjectURL(blob));
  setSelectedType(saved.chart_type);
}
```

Fix 2: Legend color doesn't match bars

The legend shows green (teal #0D9488) for all
series but bars show multiple colors. This happens
because Chart.js legend reads the first dataset's
backgroundColor[0] when it's an array.

For single-dataset charts (bar, horizontal_bar, line),

B) Semantic colors for movement/comparison charts:

Add a color scheme based on label content.
In chart-renderer.ts, after building labels, check
if this looks like a movement chart and apply
semantic colors:

```typescript
function getSemanticColors(
  labels: string[],
  defaultPalette: string[]
): string[] {
  const SEMANTIC: Record<string, string> = {
    // Movement semantics
    'created':   '#0D9488',  // teal — new/positive
    'advanced':  '#0D9488',  // teal — positive
    'won':       '#16A34A',  // green — won
    'closed won':'#16A34A',
    'regressed': '#F59E0B',  // amber — caution
    'lost':      '#EF4444',  // red — negative
    'closed lost':'#EF4444',
    // Coverage semantics
    'target':    '#CBD5E1',  // muted — benchmark
    'coverage':  '#0D9488',  // teal — actual
    'gap':       '#F59E0B',  // amber — shortfall
    // Forecast semantics
    'bear':      '#94A3B8',  // muted
    'base':      '#CBD5E1',  // light
    'bull':      '#F59E0B',  // amber — upside
    'closed-won':'#0D9488',  // teal
    'open pipeline': '#CBD5E1',
    'best case': '#F59E0B',
  };

  return labels.map((label, i) => {
    const key = label.toLowerCase().trim();
    return SEMANTIC[key] || defaultPalette[i % defaultPalette.length];
  });
}
```

Call this instead of the plain palette mapping:
```typescript
// Replace:
const colors = chart.data.map(
  (d, i) => d.color || PALETTE[i % PALETTE.length]
);

// With:
const colors = chart.data.map(
  (d, i) => d.color || getSemanticColors(labels, PALETTE)[i]
);
```
